## `Kiln.Oban.BaseWorker` — macro

```elixir
defmodule Kiln.Oban.BaseWorker do
  @moduledoc """
  Base for every Kiln Oban worker. Ships with:
    - max_attempts: 3 (PITFALLS P9 — Oban defaults to 20, too high)
    - unique config for idempotency_key-scoped dedupe
    - fetch_or_record_intent/2, complete_op/2 helpers
  """

  defmacro __using__(opts) do
    opts =
      opts
      |> Keyword.put_new(:max_attempts, 3)
      |> Keyword.put_new(:unique,
        keys: [:idempotency_key],
        period: :infinity,
        states: [:available, :scheduled, :executing]
      )

    quote do
      use Oban.Worker, unquote(opts)

      alias Kiln.ExternalOperations

      @spec fetch_or_record_intent(String.t(), map()) ::
              {:inserted_new, ExternalOperations.Operation.t()}
              | {:found_existing, ExternalOperations.Operation.t()}
              | {:error, term()}
      def fetch_or_record_intent(key, attrs) do
        ExternalOperations.fetch_or_record_intent(key, attrs)
      end

      @spec complete_op(ExternalOperations.Operation.t(), map()) ::
              {:ok, ExternalOperations.Operation.t()} | {:error, term()}
      def complete_op(op, result), do: ExternalOperations.complete_op(op, result)

      @spec fail_op(ExternalOperations.Operation.t(), map()) ::
              {:ok, ExternalOperations.Operation.t()} | {:error, term()}
      def fail_op(op, error), do: ExternalOperations.fail_op(op, error)
    end
  end
end
```

| Boundary | Description |
|----------|-------------|
| Oban worker retry ↔ External side-effect | Worker may crash between intent insert and action; retry MUST read back intent row and not re-do completed action. |

|-----------|----------|-----------|-------------|-----------------|

```elixir
defmodule Kiln.ExternalOperations.Operation do
  use Ecto.Schema
  import Ecto.Changeset

  @primary_key {:id, :binary_id, autogenerate: false}
  @foreign_key_type :binary_id

  @states [:intent_recorded, :action_in_flight, :completed, :failed, :abandoned]

  @derive {Jason.Encoder, only: [:id, :op_kind, :idempotency_key, :state, :schema_version,
                                  :intent_payload, :result_payload, :attempts,
                                  :run_id, :stage_id,
                                  :intent_recorded_at, :action_started_at, :completed_at]}

  schema "external_operations" do
    field :op_kind, :string
    field :idempotency_key, :string
    field :state, Ecto.Enum, values: @states, default: :intent_recorded
    field :schema_version, :integer, default: 1
    field :intent_payload, :map, default: %{}
    field :result_payload, :map
    field :attempts, :integer, default: 0
    field :last_error, :map
    field :run_id, :binary_id
    field :stage_id, :binary_id
    field :intent_recorded_at, :utc_datetime_usec
    field :action_started_at, :utc_datetime_usec
    field :completed_at, :utc_datetime_usec
    timestamps()
  end

  @required [:op_kind, :idempotency_key]
  @optional [:state, :schema_version, :intent_payload, :result_payload, :attempts,
             :last_error, :run_id, :stage_id,
             :intent_recorded_at, :action_started_at, :completed_at]

  def changeset(op, attrs) do
    op
    |> cast(attrs, @required ++ @optional)
    |> validate_required(@required)
    |> validate_inclusion(:state, @states)
    |> unique_constraint(:idempotency_key, name: :external_operations_idempotency_key_idx)
```

### `lib/kiln/external_operations.ex`:

```elixir
defmodule Kiln.ExternalOperations do
  @moduledoc """
  Two-phase intent → action → completion idempotency table. Every external
  side-effect (LLM, git, docker, notify, secrets) funnels through here.
  Every row mutation writes a companion audit event in the same transaction
  (D-18).
  """

  alias Kiln.Audit
  alias Kiln.ExternalOperations.Operation
  alias Kiln.Repo

  @doc """
  Insert a new intent row or return the existing one for the given idempotency_key.

  Returns:
    - `{:inserted_new, op}` if this is the first call for the key
    - `{:found_existing, op}` if the key already has a row
    - `{:error, changeset}` on validation failure

  On success: an `external_op_intent_recorded` audit event is appended in
  the same transaction (D-18).
  """
  @spec fetch_or_record_intent(String.t(), map()) ::
          {:inserted_new, Operation.t()}
          | {:found_existing, Operation.t()}
          | {:error, term()}
  def fetch_or_record_intent(idempotency_key, attrs) when is_binary(idempotency_key) do
    now = DateTime.utc_now()
    cid = attrs[:correlation_id] || Logger.metadata()[:correlation_id] || Ecto.UUID.generate()

    insert_attrs =
      attrs
      |> Map.put(:idempotency_key, idempotency_key)
      |> Map.put(:state, :intent_recorded)
      |> Map.put(:intent_recorded_at, now)

    Repo.transaction(fn ->
      case Repo.insert(Operation.changeset(%Operation{}, insert_attrs),
             on_conflict: :nothing,
             conflict_target: :idempotency_key
           ) do
        {:ok, %Operation{id: nil}} ->
          # on_conflict: :nothing with RETURNING may give a %Operation{} with nil id
          # when the row already existed. Re-read with FOR UPDATE for consistency.
          op = Repo.one!(
            from o in Operation,
              where: o.idempotency_key == ^idempotency_key,
              lock: "FOR UPDATE"
          )
          {:found_existing, op}

        {:ok, %Operation{} = op} ->
          # New row inserted; append matching audit event
          {:ok, _ev} = Audit.append(%{
            event_kind: :external_op_intent_recorded,
            run_id: op.run_id,
            stage_id: op.stage_id,
            correlation_id: cid,
            payload: %{
              "op_kind" => op.op_kind,
              "idempotency_key" => op.idempotency_key
            }
          })

          {:inserted_new, op}

        {:error, changeset} ->
          Repo.rollback(changeset)
      end
    end)
    |> case do
      {:ok, result} -> result
      {:error, reason} -> {:error, reason}
    end
  end

  @doc "Transition an op to :completed and append an audit event atomically."
  @spec complete_op(Operation.t(), map()) :: {:ok, Operation.t()} | {:error, term()}
  def complete_op(%Operation{} = op, result_payload) when is_map(result_payload) do
    cid = Logger.metadata()[:correlation_id] || Ecto.UUID.generate()

    Repo.transaction(fn ->
      changeset = Operation.changeset(op, %{
        state: :completed,
        result_payload: result_payload,
        completed_at: DateTime.utc_now()
      })

      case Repo.update(changeset) do
        {:ok, updated} ->
          {:ok, _ev} = Audit.append(%{
            event_kind: :external_op_completed,
            run_id: updated.run_id,
            stage_id: updated.stage_id,
            correlation_id: cid,
            payload: %{
              "op_kind" => updated.op_kind,
              "idempotency_key" => updated.idempotency_key,
              "result_summary" => summarize(result_payload)
            }
          })
          updated

        {:error, cs} -> Repo.rollback(cs)
      end
    end)
  end

  @doc "Transition an op to :failed and append an audit event atomically."
  @spec fail_op(Operation.t(), map()) :: {:ok, Operation.t()} | {:error, term()}
  def fail_op(%Operation{} = op, error_map) when is_map(error_map) do
    cid = Logger.metadata()[:correlation_id] || Ecto.UUID.generate()

    Repo.transaction(fn ->
      changeset = Operation.changeset(op, %{
        state: :failed,
        last_error: error_map,
        attempts: op.attempts + 1
      })

      case Repo.update(changeset) do
        {:ok, updated} ->
          {:ok, _ev} = Audit.append(%{
            event_kind: :external_op_failed,
            run_id: updated.run_id,
            stage_id: updated.stage_id,
            correlation_id: cid,
            payload: %{
              "op_kind" => updated.op_kind,
              "idempotency_key" => updated.idempotency_key,
              "error" => Jason.encode!(error_map)
            }
          })
          updated

        {:error, cs} -> Repo.rollback(cs)
      end
    end)
  end

  @doc "Mark an op as :abandoned (orphaned — intent without action). Used by StuckDetector (Phase 5)."
  @spec abandon_op(Operation.t(), String.t()) :: {:ok, Operation.t()} | {:error, term()}
  def abandon_op(%Operation{} = op, reason) do
    cid = Logger.metadata()[:correlation_id] || Ecto.UUID.generate()

    Repo.transaction(fn ->
      changeset = Operation.changeset(op, %{
        state: :abandoned,
        last_error: %{"reason" => reason}
      })

      case Repo.update(changeset) do
        {:ok, updated} ->
          {:ok, _ev} = Audit.append(%{
            event_kind: :external_op_failed,
            run_id: updated.run_id,
            stage_id: updated.stage_id,
            correlation_id: cid,
            payload: %{
              "op_kind" => updated.op_kind,
              "idempotency_key" => updated.idempotency_key,
              "error" => "abandoned: #{reason}"
            }
          })
          updated

        {:error, cs} -> Repo.rollback(cs)
      end
    end)
```

```elixir
  @moduledoc """
  Periodic Oban worker that deletes completed external_operations rows older
```

```elixir
defmodule Kiln.Oban.BaseWorker do
  @moduledoc """
  Base for every Kiln Oban worker. Injects safe defaults and idempotency helpers.

  Usage:
      defmodule MyApp.DoThingWorker do
        use Kiln.Oban.BaseWorker, queue: :default  # inherits max_attempts: 3 + unique

        @impl Oban.Worker
        def perform(%Oban.Job{args: %{"idempotency_key" => key} = args}) do
          case fetch_or_record_intent(key, %{op_kind: "do_thing", intent_payload: args}) do
            {:found_existing, %{state: :completed} = op} -> {:ok, op}
            {_whatever, op} ->
              # ... do the action idempotently ...
              complete_op(op, %{"result" => "ok"})
          end
        end
      end

  Rationale:
    - `max_attempts: 3` caps retries tightly (PITFALLS P9; Oban default is 20).
    - `unique: [keys: [:idempotency_key], period: :infinity, states: [...]]`
      enforces at-most-one-in-flight dedupe for the same key.
    - Workers may override any option in `use Kiln.Oban.BaseWorker, opts`.
  """

  defmacro __using__(opts) do
    opts =
      opts
      |> Keyword.put_new(:max_attempts, 3)
      |> Keyword.put_new(:unique,
        keys: [:idempotency_key],
        period: :infinity,
        states: [:available, :scheduled, :executing]
      )

    quote do
      use Oban.Worker, unquote(opts)

      alias Kiln.ExternalOperations

      def fetch_or_record_intent(key, attrs),
        do: ExternalOperations.fetch_or_record_intent(key, attrs)

      def complete_op(op, result), do: ExternalOperations.complete_op(op, result)

      def fail_op(op, err), do: ExternalOperations.fail_op(op, err)
    end
  end
end
```
