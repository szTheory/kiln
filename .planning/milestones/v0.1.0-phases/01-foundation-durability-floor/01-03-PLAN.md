## `Kiln.Audit.append/1` — boundary API

```elixir
@spec append(map()) :: {:ok, Kiln.Audit.Event.t()} | {:error, term()}
def append(attrs)
```

- Validates `event_kind` is in the 22-value taxonomy.
- Loads `priv/audit_schemas/v1/{event_kind}.json` via JSV.
- Validates `attrs.payload` against the schema.
- If invalid: `{:error, {:audit_payload_invalid, details}}` with no DB write.
- If valid: INSERT into `audit_events`, returns `{:ok, %Kiln.Audit.Event{}}`.
- All inserts carry `correlation_id` from `Logger.metadata[:correlation_id]` or the `attrs` map (explicit wins).
- `schema_version: 1` default; callers can override.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Runtime role ↔ audit_events | `kiln_app` cannot UPDATE/DELETE/TRUNCATE audit_events (REVOKE). Any attempt raises SQLSTATE 42501. |
| Migration role ↔ audit_events | `kiln_owner` could theoretically mutate; the BEFORE trigger raises to block migrations that forget to DISABLE TRIGGER. |
| Triggers-disabled path ↔ audit_events | If triggers are globally disabled, the RULE intercepts UPDATE/DELETE and converts to DO INSTEAD NOTHING (silent no-op; detected by row count comparison). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01 | Tampering | audit_events mutation by future migration or accidental query | mitigate | Three-layer enforcement per D-12. Migration test (this plan) asserts all three independently: (1) as `kiln_app`: `Postgrex.Error` SQLSTATE `42501`; (2) as `kiln_owner`: trigger raises with `"audit_events is append-only"`; (3) with triggers disabled: `num_rows: 0` from UPDATE, row unchanged. |
| T-03 | Elevation of Privilege | Runtime role with DDL or UPDATE on audit_events | mitigate | Two-role split; `kiln_app` explicitly `REVOKE UPDATE, DELETE, TRUNCATE ON audit_events`. Migration `CREATE ROLE kiln_owner` + `CREATE ROLE kiln_app`; GRANT only what's needed. Non-audit tables get full DML to `kiln_app` (later plans add those tables). |
| T-02 (reinforced) | Information Disclosure | Runtime env var reads | mitigate | `KILN_DB_ROLE` read only in `config/runtime.exs`; `mix check_no_compile_time_secrets` from Plan 02 blocks drift. |

```elixir
  @doc """
  Creates the two-role access model (D-48). Both roles are LOGIN-able in dev/test
  because Postgrex connects as one of them per KILN_DB_ROLE. In prod we'd constrain
  further; v1 is solo-local.
  """
  def up do
    execute """
    DO $$
    BEGIN
      IF NOT EXISTS (SELECT 1 FROM pg_roles WHERE rolname = 'kiln_owner') THEN
        CREATE ROLE kiln_owner WITH LOGIN PASSWORD 'kiln_dev_owner' CREATEDB;
      END IF;
      IF NOT EXISTS (SELECT 1 FROM pg_roles WHERE rolname = 'kiln_app') THEN
        CREATE ROLE kiln_app WITH LOGIN PASSWORD 'kiln_dev_app' NOCREATEDB;
      END IF;
    END
    $$ LANGUAGE plpgsql;
    """

    # The connecting superuser (default 'kiln' from compose.yaml) hands the database over:
    execute "GRANT CONNECT ON DATABASE kiln_dev TO kiln_owner, kiln_app"
    execute "GRANT USAGE ON SCHEMA public TO kiln_owner, kiln_app"
    # kiln_owner gets CREATE (DDL); kiln_app does not.
    execute "GRANT CREATE ON SCHEMA public TO kiln_owner"
    # Subsequent migrations (run as kiln_owner via KILN_DB_ROLE=kiln_owner) own new tables.
    # kiln_app permissions on specific tables are granted per-migration.
  end

  def down do
    execute "REVOKE CONNECT ON DATABASE kiln_dev FROM kiln_owner, kiln_app"
    execute "DROP ROLE IF EXISTS kiln_app"
    execute "DROP ROLE IF EXISTS kiln_owner"
  end
end
```

```elixir
defmodule Kiln.Repo.Migrations.CreateAuditEvents do
  use Ecto.Migration

  # Mirror Kiln.Audit.EventKind.values_as_strings/0 at migration time.
  # The Elixir module is the SSOT; this migration calls into it so the CHECK stays locked.
  @event_kinds Enum.map(Kiln.Audit.EventKind.values(), &Atom.to_string/1)

  def up do
    create table(:audit_events, primary_key: false) do
      add :id, :binary_id, primary_key: true, default: fragment("uuid_generate_v7()"), null: false
      add :event_kind, :text, null: false
      add :actor_id, :text
      add :actor_role, :text
      add :run_id, :binary_id
      add :stage_id, :binary_id
      add :correlation_id, :binary_id, null: false
      add :causation_id, :binary_id
      add :schema_version, :integer, null: false, default: 1
      add :payload, :map, null: false, default: %{}
      add :occurred_at, :utc_datetime_usec, null: false, default: fragment("now()")
      timestamps(updated_at: false)
    end

    # CHECK constraint: event_kind in the 22-value taxonomy (D-07, D-08)
    kinds_list = @event_kinds |> Enum.map(&"'#{&1}'") |> Enum.join(", ")
    execute """
    ALTER TABLE audit_events
      ADD CONSTRAINT audit_events_event_kind_check
      CHECK (event_kind IN (#{kinds_list}))
    """

    # 5 b-tree composite indexes per D-10
    create index(:audit_events, [:run_id, {:desc, :occurred_at}],
             where: "run_id IS NOT NULL",
             name: :audit_events_run_id_occurred_at_idx)

    create index(:audit_events, [:stage_id, {:desc, :occurred_at}],
             where: "stage_id IS NOT NULL",
             name: :audit_events_stage_id_occurred_at_idx)

    create index(:audit_events, [:event_kind, {:desc, :occurred_at}],
             name: :audit_events_event_kind_occurred_at_idx)

    create index(:audit_events, [:actor_id, {:desc, :occurred_at}],
             name: :audit_events_actor_id_occurred_at_idx)

    create index(:audit_events, [:correlation_id],
             name: :audit_events_correlation_id_idx)

    # Grant INSERT, SELECT to kiln_app (no UPDATE, DELETE, TRUNCATE — D-12 Layer 1)
    execute "GRANT INSERT, SELECT ON audit_events TO kiln_app"
    # Explicit REVOKE for defense in depth (even though grants above are the only ones issued)
    execute "REVOKE UPDATE, DELETE, TRUNCATE ON audit_events FROM kiln_app"
  end

  def down do
    execute "REVOKE INSERT, SELECT ON audit_events FROM kiln_app"
    drop table(:audit_events)
  end
end
```

### Migration 4: `priv/repo/migrations/20260418000004_audit_events_immutability.exs` (D-12 Layers 2 + 3)

```elixir
defmodule Kiln.Repo.Migrations.AuditEventsImmutability do
  use Ecto.Migration

  def up do
    # Layer 2: BEFORE UPDATE OR DELETE OR TRUNCATE trigger (role-bypass-resistant)
    execute """
    CREATE OR REPLACE FUNCTION audit_events_immutable()
    RETURNS trigger
    LANGUAGE plpgsql
    AS $$
    BEGIN
      RAISE EXCEPTION 'audit_events is append-only (Kiln immutability invariant); attempted % blocked', TG_OP
        USING ERRCODE = '23514';  -- check_violation (semantically fits)
    END;
    $$
    """

    execute """
    CREATE TRIGGER audit_events_no_update
      BEFORE UPDATE ON audit_events
      FOR EACH ROW EXECUTE FUNCTION audit_events_immutable()
    """

    execute """
    CREATE TRIGGER audit_events_no_delete
      BEFORE DELETE ON audit_events
      FOR EACH ROW EXECUTE FUNCTION audit_events_immutable()
    """

    # TRUNCATE triggers are statement-level (FOR EACH STATEMENT required)
    execute """
    CREATE TRIGGER audit_events_no_truncate
      BEFORE TRUNCATE ON audit_events
      FOR EACH STATEMENT EXECUTE FUNCTION audit_events_immutable()
    """

    # Layer 3: RULE DO INSTEAD NOTHING (final no-op safety net; D-12 Layer 3)
    execute "CREATE RULE audit_events_no_update_rule AS ON UPDATE TO audit_events DO INSTEAD NOTHING"
    execute "CREATE RULE audit_events_no_delete_rule AS ON DELETE TO audit_events DO INSTEAD NOTHING"
  end

  def down do
    execute "DROP RULE IF EXISTS audit_events_no_delete_rule ON audit_events"
    execute "DROP RULE IF EXISTS audit_events_no_update_rule ON audit_events"
    execute "DROP TRIGGER IF EXISTS audit_events_no_truncate ON audit_events"
    execute "DROP TRIGGER IF EXISTS audit_events_no_delete ON audit_events"
    execute "DROP TRIGGER IF EXISTS audit_events_no_update ON audit_events"
    execute "DROP FUNCTION IF EXISTS audit_events_immutable"
  end
end
```

### `lib/kiln/audit/schema_registry.ex` — loads and caches the 22 JSV schemas once per BEAM boot:

```elixir
defmodule Kiln.Audit.SchemaRegistry do
  @moduledoc """
  Loads per-kind JSON schemas from priv/audit_schemas/v1/ and resolves them
  via JSV at module load time. Cached in @schemas at compile time so runtime
  has zero file IO per append/1 call.
  """

  @priv_dir Application.compile_env(:kiln, :audit_schemas_dir, "priv/audit_schemas/v1")

  @schemas (
    for kind <- Kiln.Audit.EventKind.values(), into: %{} do
      file = Path.join([File.cwd!(), @priv_dir, "#{kind}.json"])

      case File.read(file) do
        {:ok, json} ->
          raw = Jason.decode!(json)
          root = JSV.build!(raw, resolver: JSV.Resolver.Embedded)
          {kind, root}

        {:error, :enoent} ->
          # Graceful build — Task 3 may run before all 22 files exist. Use a
          # permissive catchall at compile and upgrade to strict when all files exist.
          {kind, :missing}
      end
    end
  )

  @spec fetch(atom()) :: {:ok, JSV.Root.t()} | {:error, :schema_missing}
  def fetch(kind) when is_atom(kind) do
    case Map.get(@schemas, kind, :missing) do
      :missing -> {:error, :schema_missing}
      root -> {:ok, root}
    end
  end

  @spec loaded_kinds() :: [atom()]
  def loaded_kinds, do: @schemas |> Enum.reject(fn {_k, v} -> v == :missing end) |> Enum.map(&elem(&1, 0))
end
```

```elixir
defmodule Kiln.Audit do
  @moduledoc """
  Append-only audit ledger. The ONLY sanctioned entry point is `append/1`.
  Three DB-level layers reject mutation attempts (see D-12 / migration 20260418000004).
  """

  import Ecto.Query
  alias Kiln.Audit.{Event, EventKind, SchemaRegistry}
  alias Kiln.Repo

  @spec append(map()) :: {:ok, Event.t()} | {:error, {:audit_payload_invalid, term()} | term()}
  def append(%{event_kind: kind} = attrs) when is_atom(kind) do
    with true <- EventKind.valid?(kind) || {:error, {:unknown_event_kind, kind}},
         {:ok, root} <- SchemaRegistry.fetch(kind),
         :ok <- validate_payload(root, Map.get(attrs, :payload, %{})) do
      do_insert(attrs)
    else
      {:error, :schema_missing} ->
        {:error, {:audit_schema_missing, kind}}

      {:error, reason} ->
        {:error, reason}
    end
  end

  def append(%{event_kind: kind_str} = attrs) when is_binary(kind_str) do
    case Enum.find(EventKind.values(), &(Atom.to_string(&1) == kind_str)) do
      nil -> {:error, {:unknown_event_kind, kind_str}}
      kind -> append(Map.put(attrs, :event_kind, kind))
    end
  end

  @spec replay(keyword()) :: [Event.t()]
  def replay(opts \\ []) do
    q = from(e in Event, order_by: [asc: e.occurred_at])

    q =
      Enum.reduce(opts, q, fn
        {:run_id, rid}, acc -> where(acc, [e], e.run_id == ^rid)
        {:event_kind, k}, acc -> where(acc, [e], e.event_kind == ^k)
        {:correlation_id, c}, acc -> where(acc, [e], e.correlation_id == ^c)
        _other, acc -> acc
      end)

    Repo.all(q)
  end

  defp validate_payload(root, payload) do
    case JSV.validate(payload, root) do
      {:ok, _cast} -> :ok
      {:error, errors} -> {:error, {:audit_payload_invalid, errors}}
    end
  end

  defp do_insert(attrs) do
    # Pull correlation_id from Logger.metadata if not explicit (D-46)
    cid = attrs[:correlation_id] || Logger.metadata()[:correlation_id] ||
            raise ArgumentError, "Kiln.Audit.append/1 requires correlation_id (explicit or via Logger.metadata)"

    attrs =
      attrs
      |> Map.put_new(:correlation_id, cid)
      |> Map.put_new(:schema_version, 1)
      |> Map.put_new(:occurred_at, DateTime.utc_now())

    %Event{}
    |> Ecto.Changeset.cast(attrs, [
      :event_kind, :actor_id, :actor_role, :run_id, :stage_id,
      :correlation_id, :causation_id, :schema_version, :payload, :occurred_at
    ])
    |> Ecto.Changeset.validate_required([:event_kind, :correlation_id])
    |> Repo.insert()
  end
end
```
