```elixir
defmodule Kiln.Credo.NoProcessPut do
  use Credo.Check,
    base_priority: :high,
    category: :warning,
    explanations: [
      check: """
      `Process.put/1,2` is banned by the Kiln project (CLAUDE.md → Conventions →
      Elixir-specific anti-patterns). Use explicit threading or function args
      for state; use `Kiln.Telemetry.pack_ctx/0` / `unpack_ctx/1` for cross-process
      context propagation.
      """
    ]

  @impl true
  def run(%SourceFile{} = source_file, params \\ []) do
    issue_meta = IssueMeta.for(source_file, params)
    Credo.Code.prewalk(source_file, &traverse(&1, &2, issue_meta))
  end

  defp traverse({{:., _, [{:__aliases__, _, [:Process]}, :put]}, meta, _args} = ast, issues, issue_meta) do
    {ast, [issue_for(issue_meta, meta[:line]) | issues]}
  end

  defp traverse(ast, issues, _), do: {ast, issues}

  defp issue_for(issue_meta, line) do
    format_issue(issue_meta,
      message: "Process.put/* is banned — use explicit threading (Kiln.Telemetry.pack_ctx/0).",
      line_no: line
    )
  end
end
```

## `mix check_no_compile_time_secrets` — Mix task (D-26, T-02)

Greps `config/config.exs`, `config/dev.exs`, `config/prod.exs`, `config/test.exs` for `System.get_env` or `System.fetch_env!` and raises if found. Allow them in `config/runtime.exs`.

```elixir
defmodule Mix.Tasks.CheckNoCompileTimeSecrets do
  use Mix.Task
  @shortdoc "Fail if config/*.exs (non-runtime) reads env vars (T-02)."

  @impl true
  def run(_args) do
    files = ~w(config/config.exs config/dev.exs config/prod.exs config/test.exs)
    offenders =
      files
      |> Enum.filter(&File.exists?/1)
      |> Enum.flat_map(fn f ->
        f
        |> File.stream!()
        |> Enum.with_index(1)
        |> Enum.filter(fn {line, _i} ->
          String.contains?(line, "System.get_env") or
            String.contains?(line, "System.fetch_env!")
        end)
        |> Enum.map(fn {line, i} -> {f, i, String.trim(line)} end)
      end)

    case offenders do
      [] -> :ok
      list ->
        list
        |> Enum.map(fn {f, i, l} -> "  #{f}:#{i}  #{l}" end)
        |> Enum.join("\n")
        |> then(&Mix.raise("Compile-time secret read detected (move to config/runtime.exs):\n#{&1}"))
    end
  end
end
```

```elixir
defmodule Kiln.Credo.NoMixEnvAtRuntime do
  @moduledoc false
  use Credo.Check,
    base_priority: :high,
    category: :warning,
    explanations: [
      check: """
      `Mix.env()` at runtime is banned (CLAUDE.md Elixir anti-patterns). Mix is
      unavailable in release builds — `Mix.env/0` at runtime raises
      `UndefinedFunctionError`. Use `Application.get_env(:kiln, :env)`, set from
      `config/*.exs`, or runtime-config flags in `config/runtime.exs`.
      """
    ]

  alias Credo.Code
  alias Credo.IssueMeta

  @impl true
  def run(%Credo.SourceFile{filename: filename} = source_file, params \\ []) do
    # Allow Mix.env() inside the mix.exs project file (compile-time project config).
    if Path.basename(filename) == "mix.exs" do
      []
    else
      issue_meta = IssueMeta.for(source_file, params)
      Code.prewalk(source_file, &traverse(&1, &2, issue_meta))
    end
  end

  defp traverse({{:., _, [{:__aliases__, _, [:Mix]}, :env]}, meta, _args} = ast, issues, issue_meta) do
    {ast, [issue_for(issue_meta, meta[:line]) | issues]}
  end

  defp traverse(ast, issues, _), do: {ast, issues}

  defp issue_for(issue_meta, line) do
    format_issue(issue_meta,
      message: "Mix.env/0 is unavailable in releases — use Application.get_env(:kiln, :env) instead.",
      line_no: line
    )
  end
end
```
