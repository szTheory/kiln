defmodule Kiln.Runs.Transitions do
  import Ecto.Query
  require Logger
  alias Kiln.{Audit, Repo}
  alias Kiln.Runs.Run
  alias Kiln.Policies.StuckDetector

  @terminal ~w(merged failed escalated)a
  @any_state ~w(queued planning coding testing verifying blocked)a

  # D-87 — matrix encoded as data, NOT pattern-matched function heads.
  @matrix %{
    queued:    [:planning],
    planning:  [:coding, :blocked],
    coding:    [:testing, :blocked, :planning],
    testing:   [:verifying, :blocked, :planning],
    verifying: [:merged, :planning, :blocked],
    blocked:   [:planning, :coding, :testing, :verifying]
  }

  # Every non-terminal state can reach :escalated or :failed (cross-cutting)
  @cross_cutting ~w(escalated failed)a

  @spec transition(Ecto.UUID.t(), atom(), map()) ::
          {:ok, Run.t()} | {:error, :illegal_transition | :not_found | term()}
  def transition(run_id, to, meta \\ %{}) when is_atom(to) do
    result =
      Repo.transact(fn ->
        with {:ok, run} <- lock_run(run_id),
             :ok <- assert_allowed(run.state, to),
             :ok <- StuckDetector.check(%{run: run, to: to, meta: meta}),
             {:ok, updated} <- update_state(run, to, meta),
             {:ok, _event} <- append_audit(updated, run.state, to, meta) do
          {:ok, updated}
        end
      end)

    # CRITICAL: PubSub broadcast AFTER tx commits — never inside the closure.
    case result do
      {:ok, run} ->
        Phoenix.PubSub.broadcast(Kiln.PubSub, "run:#{run.id}", {:run_state, run})
        Phoenix.PubSub.broadcast(Kiln.PubSub, "runs:board", {:run_state, run})
        {:ok, run}
      other ->
        other
    end
  end

  @spec transition!(Ecto.UUID.t(), atom(), map()) :: Run.t()
  def transition!(run_id, to, meta \\ %{}) do
    case transition(run_id, to, meta) do
      {:ok, run} -> run
      {:error, :illegal_transition} ->
        run = Repo.get(Run, run_id)
        raise Kiln.Runs.IllegalTransitionError,
          run_id: run_id,
          from: run && run.state,
          to: to,
          allowed: Map.get(@matrix, run && run.state, []) ++ @cross_cutting
      {:error, :not_found} ->
        raise Kiln.Runs.IllegalTransitionError,
          run_id: run_id,
          from: :not_found,
          to: to,
          allowed: []
      {:error, reason} ->
        raise "Kiln.Runs.Transitions.transition!/3 error: #{inspect(reason)}"
    end
  end

  @spec matrix() :: %{atom() => [atom()]}
  def matrix, do: @matrix

  # -- private helpers --

  defp lock_run(run_id) do
    case Repo.one(from(r in Run, where: r.id == ^run_id, lock: "FOR UPDATE")) do
      nil -> {:error, :not_found}
      run -> {:ok, run}
    end
  end

  defp assert_allowed(from, to) do
    cond do
      from in @terminal -> {:error, :illegal_transition}
      to in (Map.get(@matrix, from, []) ++ @cross_cutting) -> :ok
      true -> {:error, :illegal_transition}
    end
  end

  defp update_state(run, to, meta) do
    attrs = %{state: to}
    attrs =
      case meta do
        %{reason: r} when to in @cross_cutting -> Map.put(attrs, :escalation_reason, Atom.to_string(r))
        _ -> attrs
      end

    run
    |> Run.transition_changeset(attrs, meta)
    |> Repo.update()
  end

  defp append_audit(run, from, to, meta) do
    payload =
      %{"from" => Atom.to_string(from), "to" => Atom.to_string(to)}
      |> maybe_add_reason(meta)

    Audit.append(%{
      event_kind: :run_state_transitioned,
      run_id: run.id,
      correlation_id: Logger.metadata()[:correlation_id] || Ecto.UUID.generate(),
      payload: payload
    })
  end

  defp maybe_add_reason(payload, %{reason: r}) when is_atom(r),
    do: Map.put(payload, "reason", Atom.to_string(r))
  defp maybe_add_reason(payload, _), do: payload
end

defmodule Kiln.Policies.StuckDetector do
  use GenServer

  @spec start_link(keyword()) :: GenServer.on_start()
  def start_link(opts), do: GenServer.start_link(__MODULE__, opts, name: __MODULE__)

  @spec check(map()) :: :ok | {:halt, atom(), map()}
  def check(ctx) when is_map(ctx), do: GenServer.call(__MODULE__, {:check, ctx})

  @impl true
  def init(_opts), do: {:ok, %{}}

  @impl true
  def handle_call({:check, _ctx}, _from, state) do
    # P2: no-op — the hook path IS the behavior to exercise. P5 fills the
    # sliding-window body. Stable signature: :ok | {:halt, reason, payload}.
    {:reply, :ok, state}
  end
end

defmodule Kiln.Runs.IllegalTransitionError do
  defexception [:run_id, :from, :to, :allowed, :message]

  @impl true
  def exception(fields) do
    fields = Keyword.new(fields)
    run_id = Keyword.get(fields, :run_id)
    from = Keyword.get(fields, :from)
    to = Keyword.get(fields, :to)
    allowed = Keyword.get(fields, :allowed, [])

    msg = "illegal run state transition for run_id=#{inspect(run_id)}: " <>
          "from #{inspect(from)} to #{inspect(to)}; " <>
          "allowed from #{inspect(from)}: #{inspect(allowed)}"

    struct!(__MODULE__, Keyword.put(fields, :message, msg))
  end
end

## Threats
- **T1: Audit event forgery via direct Repo.update bypassing transition/3** (severity: medium)
  - Vector: A developer calls `Run.changeset(run, %{state: :merged}) |> Repo.update()` directly — bypasses assertion + audit append. Run ends up in `:merged` with no audit trail.
  - Mitigation: Kiln.Runs.Transitions is documented as the sole state-mutation path. Phase 1 D-12's three-layer enforcement guarantees audit events are INSERT-only (no tampering post-hoc), but it does NOT force callers through transition/3. Mitigation at dev-time: a Credo check like `NoDirectRunStateUpdate` (scans for `Run.transition_changeset` callers outside `Kiln.Runs.Transitions`) could catch drift — scope to a Phase 3+ hardening plan, not P2. Documented in `lib/kiln/runs/transitions.ex` @moduledoc.
  - Residual risk: Drift until the custom Credo check ships in a later phase; caught by PR review in the interim.

- **T2: SELECT ... FOR UPDATE lock-wait deadlock under high concurrency** (severity: low)
  - Vector: Two transitions on the same run lock-wait → second blocks until first commits. With 4 stages queue × 10 concurrent runs, lock contention is possible.
  - Mitigation: `pool_size: 20` (Plan 04) gives 4 connections of headroom beyond the 16 Oban workers. Postgres lock_timeout (default none) could be set via `config :kiln, Kiln.Repo, parameters: [lock_timeout: "30s"]` in a future phase — not required for solo-op. Concurrent-run ceiling is 10 per D-95.
  - Residual risk: A stuck transaction (e.g., test that didn't release) leaks a lock — caught by Postgres `pg_locks` inspection.

- **T3: PubSub flood — LiveView OOM on rapid run-state churn** (severity: low)
  - Vector: A run bouncing between `coding ↔ planning` rapidly sends :run_state messages at high rate; LiveView stream buffers could grow.
  - Mitigation: Phase 7 LiveViews use bounded streams per roadmap P13 pitfall. Phase 2 doesn't ship LV consumers. PubSub topic "runs:board" is global-fanout but only active LV subscribers receive.
  - Residual risk: Phase 7 concern.

- **T4: StuckDetector hook becomes a performance bottleneck if P5 body is slow** (severity: low)
  - Vector: P5 replaces the no-op body with a sliding-window scan that could take 50ms+ under load; every transition blocks for that long.

  - Residual risk: P5 risk; documented in StuckDetector @moduledoc.
