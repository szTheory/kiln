```elixir
defmodule Kiln.Sandboxes.Limits do
  @moduledoc """
  Per-stage-kind sandbox resource limits (D-112). Loaded from
  `priv/sandbox/limits.yaml` at app start into `:persistent_term`.

  `for_stage/1` returns the limits map. Falls back to `:default` for
  unknown stage kinds (warn-log, never raise).

  Wave 5 wires `Kiln.Application.start/2` to call `load!/0` early in boot.
  """

  require Logger

  @term_key {__MODULE__, :limits_table}
  @yaml_path Path.expand("../../../priv/sandbox/limits.yaml", __DIR__)
  @external_resource @yaml_path

  @spec load! :: :ok
  def load! do
    case YamlElixir.read_from_file(@yaml_path) do
      {:ok, table} when is_map(table) ->
        :persistent_term.put(@term_key, table)
        :ok

      {:error, reason} ->
        raise "Kiln.Sandboxes.Limits could not load \#{@yaml_path}: \#{inspect(reason)}"
    end
  end

  @spec for_stage(atom() | String.t()) :: map()
  def for_stage(stage_kind) do
    table = :persistent_term.get(@term_key, %{})
    key = to_string(stage_kind)

    case Map.get(table, key) do
      nil ->
        Logger.warning("Kiln.Sandboxes.Limits: unknown stage kind #{stage_kind}, using :default")
        Map.fetch!(table, "default")

      v ->
        v
    end
  end

  @spec all_stage_kinds :: [String.t()]
  def all_stage_kinds do
    table = :persistent_term.get(@term_key, %{})
    Map.keys(table)
  end
end
```

```elixir
defmodule Mix.Tasks.Kiln.Sandbox.Build do
  use Mix.Task
  @shortdoc "Build the Kiln sandbox Docker image and update images.lock"

  @impl Mix.Task
  def run(_args) do
    image_tag = "kiln/sandbox-elixir:" <> git_sha()
    dockerfile = "priv/sandbox/elixir.Dockerfile"

    {_out, status} =
      System.cmd("docker",
        ["build", "-f", dockerfile, "-t", image_tag, "priv/sandbox"],
        into: IO.stream(:stdio, :line)
      )

    if status != 0, do: Mix.raise("docker build failed (exit \#{status})")

    {digest_out, 0} = System.cmd("docker", ["inspect", "--format={{index .RepoDigests 0}}", image_tag])
    digest = digest_out |> String.trim() |> String.split("@") |> List.last()

    new_lock = """
    # Auto-generated by `mix kiln.sandbox.build`. Commit to track image evolution.
    elixir:
      image_ref: \"\#{image_tag}\"
      image_digest: \"\#{digest}\"
      built_at: \"\#{DateTime.utc_now() |> DateTime.to_iso8601()}\"
      base_image: \"hexpm/elixir:1.19.5-erlang-28.1.1-alpine-3.21\"
    """

    File.write!("priv/sandbox/images.lock", new_lock)
    Mix.shell().info("Built #{image_tag}\nDigest: #{digest}\nWrote priv/sandbox/images.lock")
  end

  defp git_sha do
    case System.cmd("git", ["rev-parse", "--short", "HEAD"]) do
      {sha, 0} -> String.trim(sha)
      _ -> "unknown"
    end
  end
end
```

```elixir
defmodule Kiln.Sandboxes.ImageResolver do
  @moduledoc """
  Pure resolution: language atom → {image_ref, image_digest} from
  `priv/sandbox/images.lock` (D-111).

  P3 ships Elixir only; sibling Dockerfile additions for typescript /
  python / etc. happen per-adoption with no refactor.
  """

  @lock_path Path.expand("../../../priv/sandbox/images.lock", __DIR__)
  @external_resource @lock_path

  @lock_data (case File.read(@lock_path) do
                {:ok, raw} -> YamlElixir.read_from_string!(raw)
                _ -> %{}
              end)

  @spec resolve(String.t() | atom()) :: {:ok, {String.t(), String.t()}} | {:error, :unsupported_language}
  def resolve(lang) do
    key = to_string(lang || "elixir")

    case Map.get(@lock_data, key) do
      %{"image_ref" => ref, "image_digest" => digest} -> {:ok, {ref, digest}}
      _ -> {:error, :unsupported_language}
    end
  end

  @spec all_supported_languages :: [String.t()]
  def all_supported_languages, do: Map.keys(@lock_data)
end
```

```elixir
defmodule Kiln.Sandboxes.EnvBuilder do
  @moduledoc """
  Builds a per-stage `.env` file consumed by `docker run --env-file`
  (D-134 — env ALLOWLIST enforcement). Fails launch on any env var
  name matching the secret-shape regex.

  Allowed keys (P3):
    * DTU_BASE_URL — DTU sidecar base URL
    * DTU_TOKEN — short-lived (<5min) token scoped to DTU only
    * MIX_ENV — usually `test`
    * LANG — usually `en_US.UTF-8`

  Any other key whose name matches `~r/(api_key|secret|token|authorization|bearer)/i`
  (case-insensitive, with explicit DTU_TOKEN exception) returns
  `{:error, :sandbox_env_contains_secret, key_name}`. Callers
  (DockerDriver) raise `Kiln.Blockers.BlockedError{reason: :policy_violation}`.

  Files are written 0600 to `priv/run/<stage_run_id>.env` and deleted at
  end of stage-completion transaction.
  """

  @allowlist ~w(DTU_BASE_URL MIX_ENV LANG)
  @secret_regex ~r/(api_key|secret|token|authorization|bearer)/i

  @env_dir "priv/run"

  @spec build(map(), String.t()) :: {:ok, String.t()} | {:error, atom(), atom() | String.t()}
  def build(env_map, stage_run_id) when is_map(env_map) and is_binary(stage_run_id) do
    case validate(env_map) do
      :ok ->
        File.mkdir_p!(@env_dir)
        path = Path.join(@env_dir, "\#{stage_run_id}.env")
        body = render(env_map)
        File.write!(path, body, [:write])
        File.chmod!(path, 0o600)
        {:ok, path}

      {:error, _kind, _key} = err ->
        err
    end
  end

  @spec delete!(String.t()) :: :ok
  def delete!(path) when is_binary(path) do
    case File.rm(path) do
      :ok -> :ok
      {:error, :enoent} -> :ok
      {:error, reason} -> raise "Failed to delete env file #{path}: #{inspect(reason)}"
    end
  end

  defp validate(env_map) do
    Enum.reduce_while(env_map, :ok, fn {k, _v}, _acc ->
      key_str = to_string(k)

      cond do
        key_str in @allowlist ->
          {:cont, :ok}

        # DTU_TOKEN is an explicit exception (it IS a token, but it's
        # short-lived + scoped to DTU only per D-134).
        key_str == "DTU_TOKEN" ->
          {:cont, :ok}

        Regex.match?(@secret_regex, key_str) ->
          {:halt, {:error, :sandbox_env_contains_secret, key_str}}

        true ->
          {:halt, {:error, :sandbox_env_not_allowlisted, key_str}}
      end
    end)
  end

  defp render(env_map) do
    env_map
    |> Enum.map(fn {k, v} -> "\#{k}=\#{v}" end)
    |> Enum.join("\n")
  end
end
```

```elixir
defmodule Kiln.Sandboxes.Hydrator do
  @moduledoc """
  Pre-stage CAS materialization (D-113). Pure module called synchronously
  by `Kiln.Stages.StageWorker.perform/1` BEFORE the docker container
  launches. Reads declared `input_artifacts` from the stage input contract,
  fetches each artifact_ref from CAS, writes to `<workspace_dir>/<name>`.

  NOT a GenServer (PATTERNS.md anti-pattern #2). NOT supervised.
  """

  require Logger
  alias Kiln.Artifacts

  @type artifact_ref :: %{required(:name) => String.t(), required(:sha256) => String.t(), optional(:size_bytes) => non_neg_integer()}

  @spec hydrate([artifact_ref()], String.t(), keyword()) :: {:ok, [String.t()]} | {:error, term()}
  def hydrate(artifact_refs, workspace_dir, opts \\ []) when is_list(artifact_refs) and is_binary(workspace_dir) do
    File.mkdir_p!(workspace_dir)
    chunk_size = Keyword.get(opts, :chunk_size, 64 * 1024)

    artifact_refs
    |> Enum.reduce_while({:ok, []}, fn ref, {:ok, paths} ->
      target_path = Path.join(workspace_dir, ref.name)

      case stream_to_file(ref, target_path, chunk_size) do
        :ok -> {:cont, {:ok, [target_path | paths]}}
        {:error, reason} -> {:halt, {:error, reason}}
      end
    end)
    |> case do
      {:ok, paths} -> {:ok, Enum.reverse(paths)}
      err -> err
    end
  end

  defp stream_to_file(ref, target_path, _chunk_size) do
    case Artifacts.get(ref.sha256, name: ref.name) do
      {:ok, _artifact} ->
        # Use Artifacts.stream!/1 for byte streaming if available;
        # else read!/1 for small artifacts (acceptable for P3).
        try do
          stream = Artifacts.stream!(ref.sha256)
          File.write!(target_path, stream |> Enum.into(<<>>))
          :ok
        rescue
          UndefinedFunctionError ->
            # Fallback to read!/1 if stream!/1 not yet exported
            content = Artifacts.read!(ref.sha256)
            File.write!(target_path, content)
            :ok

          e -> {:error, {:hydrate_failed, e}}
        end

      {:error, :not_found} ->
        {:error, {:missing_artifact, ref.sha256}}

      err ->
        err
    end
  end
end
```

```elixir
defmodule Kiln.Sandboxes.Harvester do
  @moduledoc """
  Post-stage CAS write (D-113 / SAND-04). Pure module called synchronously
  by `Kiln.Stages.StageWorker.perform/1` AFTER container exits. Walks
  `<workspace_dir>/out/`, streams each file through `Kiln.Artifacts.put/4`
  (which writes CAS + emits `artifact_written` audit event in one tx —
  see PATTERNS §9, mirror of Plan 02-03 pattern verbatim).

  NOT a GenServer.
  """

  require Logger
  alias Kiln.Artifacts

  @type artifact_record :: %{name: String.t(), sha256: String.t(), size_bytes: non_neg_integer()}

  @spec harvest(String.t(), Ecto.UUID.t(), Ecto.UUID.t()) :: {:ok, [artifact_record()]} | {:error, term()}
  def harvest(workspace_dir, run_id, stage_run_id) when is_binary(workspace_dir) do
    out_dir = Path.join(workspace_dir, "out")

    if File.dir?(out_dir) do
      out_dir
      |> File.ls!()
      |> Enum.sort()  # deterministic order
      |> Enum.reduce_while({:ok, []}, fn name, {:ok, acc} ->
        full = Path.join(out_dir, name)

        if File.regular?(full) do
          case put_artifact(full, name, run_id, stage_run_id) do
            {:ok, record} -> {:cont, {:ok, [record | acc]}}
            err -> {:halt, err}
          end
        else
          {:cont, {:ok, acc}}  # skip subdirs in P3
        end
      end)
      |> case do
        {:ok, recs} -> {:ok, Enum.reverse(recs)}
        err -> err
      end
    else
      {:ok, []}
    end
  end

  defp put_artifact(path, name, run_id, stage_run_id) do
    body = File.stream!(path, [], 64 * 1024)
    content_type = guess_content_type(name)

    case Artifacts.put(body, name,
           stage_run_id: stage_run_id,
           run_id: run_id,
           content_type: content_type,
           producer_kind: :sandbox_harvest
         ) do
      {:ok, artifact} ->
        {:ok, %{name: name, sha256: artifact.sha256, size_bytes: artifact.size_bytes}}

      err ->
        err
    end
  end

  # NOTE: signature variation — Plan 02-03's Artifacts.put/4 takes the body
  # as a streamable + name as 2nd arg. Verify against lib/kiln/artifacts.ex
  # actual signature; adjust call site if needed.

  defp guess_content_type(name) do
    case Path.extname(name) do
      ".ex" -> "text/x-elixir"
      ".exs" -> "text/x-elixir"
      ".md" -> "text/markdown"
      ".json" -> "application/json"
      ".yaml" -> "text/yaml"
      ".yml" -> "text/yaml"
      ".txt" -> "text/plain"
      _ -> "application/octet-stream"
    end
  end
end
```
