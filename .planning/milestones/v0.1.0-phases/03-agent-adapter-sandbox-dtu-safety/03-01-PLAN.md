defmodule Kiln.Secrets.Ref do
  @moduledoc """
  A reference to a named secret held in `:persistent_term`. Raw string
  values NEVER appear on this struct — the `name` atom is resolved to
  the raw string only via `Kiln.Secrets.reveal!/1` inside the adapter's
  HTTP-call stack frame (D-132 / D-133 Layer 1).

  `@derive {Inspect, except: [:name]}` renders as `#Secret<name>` via
  the custom `Inspect.Kiln.Secrets.Ref` implementation below.

  This is D-133 Layer 2 (Elixir `@derive Inspect` boundary) of the
  six-layer redaction defense.
  """

  @derive {Inspect, except: [:name]}
  defstruct [:name]

  @type t :: %__MODULE__{name: atom()}
end

defimpl Inspect, for: Kiln.Secrets.Ref do
  def inspect(%Kiln.Secrets.Ref{name: name}, _opts) do
    "#Secret<\#{name}>"
  end
end
```

**CRITICAL:** The `@derive {Inspect, except: [:name]}` attribute alone would render `#Kiln.Secrets.Ref<...>`, leaking the module path but hiding the value — acceptable per D-133 Layer 2, but we upgrade by writing a custom `Inspect` impl that renders `#Secret<name>` directly. Prefer the explicit `defimpl` over the `@derive`-only form for DX readability.

defmodule Kiln.Secrets do
  @moduledoc """
  Reference-only secret store over `:persistent_term` (D-131).

  Semantics:
    * `put/2` — writes at boot via `config/runtime.exs`. Write-once at boot
      (no runtime mutation) per D-131 to avoid global GC thrash.
    * `get!/1` — returns `%Kiln.Secrets.Ref{}` only. Raises `KeyError`
      if the secret was never `put/2`.
    * `get/1` — non-raising variant; returns `{:ok, %Ref{}}` or `:error`.
    * `present?/1` — `true` iff `:persistent_term` has a non-nil value
      for `{Kiln.Secrets, name}`.
    * `reveal!/1` — the SOLE raw-string reveal path. Takes either a
      `%Ref{}` or a bare atom name. Raises `KeyError` on missing.

  **Grep audit target:** Phase 3 ships exactly 3 `Kiln.Secrets.reveal!/1`
  call sites — one per live-ish provider adapter's HTTP-call function
  (`Kiln.Agents.Adapter.Anthropic.call_http/2`, `.OpenAI.call_http/2`,
  `.Google.call_http/2`). Ollama scaffolded adapter is local, no key.

  D-133 Layer 1 (type system boundary): raw string ONLY crosses the
  function-stack-frame boundary of `reveal!/1`.
  """

  alias Kiln.Secrets.Ref

  @spec put(atom(), binary() | nil) :: :ok
  def put(name, nil) when is_atom(name) do
    :persistent_term.erase({__MODULE__, name})
    :ok
  end

  def put(name, value) when is_atom(name) and is_binary(value) do
    :persistent_term.put({__MODULE__, name}, value)
    :ok
  end

  @spec get!(atom()) :: Ref.t()
  def get!(name) when is_atom(name) do
    # Raise KeyError if missing — FAIL LOUDLY (CLAUDE.md).
    _ = :persistent_term.get({__MODULE__, name})
    %Ref{name: name}
  end

  @spec get(atom()) :: {:ok, Ref.t()} | :error
  def get(name) when is_atom(name) do
    case :persistent_term.get({__MODULE__, name}, :__absent__) do
      :__absent__ -> :error
      _ -> {:ok, %Ref{name: name}}
    end
  end

  @spec present?(atom()) :: boolean()
  def present?(name) when is_atom(name) do
    :persistent_term.get({__MODULE__, name}, nil) != nil
  end

  @spec reveal!(Ref.t() | atom()) :: binary()
  def reveal!(%Ref{name: name}), do: reveal!(name)

  def reveal!(name) when is_atom(name) do
    :persistent_term.get({__MODULE__, name})
  end
end
```

```elixir
defmodule Kiln.Logging.SecretRedactor do
  @moduledoc """
  `LoggerJSON.Redactor` behaviour impl (D-133 Layer 4). Scrubs log metadata
  entries whose KEY matches one of `@secret_key_substrings` (case-insensitive)
  OR whose VALUE is a binary starting with one of `@secret_value_prefixes`.

  Registered in `config/config.exs`:
      config :logger_json, :redactors, [{Kiln.Logging.SecretRedactor, []}]

  Non-secret values pass through unchanged. Non-binary values pass through
  unchanged (redaction operates on log-serializable strings only).
  """

  @behaviour LoggerJSON.Redactor

  @secret_key_substrings ~w(api_key secret token authorization bearer)
  @secret_value_prefixes ~w(sk-ant- sk-proj- ghp_ gho_ AIza)
  @redacted "**redacted**"

  @impl LoggerJSON.Redactor
  def redact(key, value, _opts) do
    cond do
      key_looks_secret?(key) -> @redacted
      value_looks_secret?(value) -> @redacted
      true -> value
    end
  end

  defp key_looks_secret?(key) do
    key_str = key_to_string(key)
    lowered = String.downcase(key_str)
    Enum.any?(@secret_key_substrings, &String.contains?(lowered, &1))
  end

  defp key_to_string(key) when is_atom(key), do: Atom.to_string(key)
  defp key_to_string(key) when is_binary(key), do: key
  defp key_to_string(_), do: ""

  defp value_looks_secret?(value) when is_binary(value) do
    Enum.any?(@secret_value_prefixes, &String.starts_with?(value, &1))
  end

  defp value_looks_secret?(_), do: false
end
```
