defmodule Kiln.Stages.NextStageDispatcher do
  @moduledoc """
  Pure module (D-144 / PATTERNS anti-pattern #2). Picks up Phase 2's
  deferred auto-enqueue responsibility. Called by `StageWorker.perform/1`
  after a successful stage completion, INSIDE the stage-completion
  Postgres transaction.

  Reads `CompiledGraph.stages_by_id` from the run's pinned workflow,
  finds stages whose `depends_on` is satisfied by current run's
  StageRun state distribution, enqueues next StageWorker Oban job(s)
  with deterministic idempotency key `"run:<run_id>:stage:<stage_id>"`
  (D-70).

  Handles:
    * Fan-out (N downstream stages with their parents satisfied → N jobs)
    * Fan-in barrier (stage not enqueued until ALL parents :completed)

  No GenServer.
  """

  import Ecto.Query
  alias Kiln.{Repo, Runs, Workflows}
  alias Kiln.Stages.{StageRun, StageWorker}

  @spec enqueue_next!(run_id :: Ecto.UUID.t(), completed_stage_id :: String.t()) :: :ok
  def enqueue_next!(run_id, completed_stage_id) when is_binary(completed_stage_id) do
    run = Runs.get!(run_id)
    {:ok, cg} = Workflows.load_compiled(run.workflow_id)

    stage_state = stage_state_map(run_id)

    cg.stages_by_id
    |> Enum.flat_map(fn {stage_id, stage} ->
      if depends_on_includes?(stage, completed_stage_id) and
         all_deps_completed?(stage, stage_state) and
         not Map.has_key?(stage_state, stage_id) do
        [{stage_id, stage}]
      else
        []
      end
    end)
    |> Enum.each(fn {stage_id, stage} -> enqueue_one(run, stage_id, stage) end)

    :ok
  end

  defp depends_on_includes?(%{depends_on: deps}, stage_id) when is_list(deps),
    do: stage_id in deps

  defp depends_on_includes?(_, _), do: false

  defp all_deps_completed?(%{depends_on: deps}, stage_state) do
    Enum.all?(deps, fn dep ->
      Map.get(stage_state, dep) == :completed
    end)
  end

  defp stage_state_map(run_id) do
    Repo.all(
      from sr in StageRun,
        where: sr.run_id == ^run_id,
        select: {sr.stage_id, sr.state}
    )
    |> Map.new()
  end

  defp enqueue_one(run, stage_id, stage) do
    args = %{
      "idempotency_key" => "run:\#{run.id}:stage:\#{stage_id}",
      "run_id" => run.id,
      "stage_id" => stage_id,
      "stage_kind" => Atom.to_string(stage.kind),
      "stage_input" => build_stage_input(run, stage)
    }

    args
    |> StageWorker.new(meta: Kiln.Telemetry.pack_meta())
    |> Oban.insert!()
  end

  defp build_stage_input(_run, _stage) do
    # Phase 2's StageWorker built input from artifacts CAS via stage.input_contract.
    # Phase 3 preserves that signature; the Oban worker re-derives input on
    # perform/1 from the stage_run record.
    %{}
  end
end

# REPLACE the existing stub_dispatch/3 call body with this:
defp dispatch_stage_p3(run_id, stage_run_id, stage_kind, stage_input, stage_run, run) do
  workspace = "/tmp/kiln-stage-#{stage_run_id}"
  File.mkdir_p!(workspace)

  with {:ok, _hydrated_paths} <- Kiln.Sandboxes.Hydrator.hydrate(stage_input["input_artifacts"] || [], workspace),
       {:ok, role, model} <- resolve_role_and_model(run, stage_kind),
       prompt = build_prompt(stage_kind, stage_input, model),
       :ok <- Kiln.Agents.BudgetGuard.check!(prompt, run_id: run_id, stage_id: stage_run.stage_id, model: model, adapter: Kiln.ModelRegistry.adapter_for(model)),
       container_spec = build_container_spec(stage_kind, stage_run_id, run_id, workspace),
       {:ok, _container_result} <- Kiln.Sandboxes.DockerDriver.run_stage(container_spec),
       adapter = Kiln.ModelRegistry.adapter_for(model),
       {:ok, _response} <- adapter.complete(prompt, run_id: run_id, stage_id: stage_run.stage_id, role: role, model: model),
       {:ok, _artifacts} <- Kiln.Sandboxes.Harvester.harvest(workspace, run_id, stage_run_id) do
    File.rm_rf!(workspace)
    {:ok, :stage_completed}
  else
    {:error, reason} ->
      File.rm_rf!(workspace)
      {:error, reason}
  end
end

defp resolve_role_and_model(run, stage_kind) do
  preset = String.to_existing_atom(run.model_profile_snapshot["preset"] || "elixir_lib")
  role = stage_kind_to_role(stage_kind)
  resolved = Kiln.ModelRegistry.resolve(preset)

  case Map.get(resolved, role) do
    %{model: model} -> {:ok, role, model}
    nil -> {:error, {:unknown_role, role}}
  end
end

defp stage_kind_to_role(:verifying), do: :qa_verifier

defp stage_kind_to_role(other), do: other
