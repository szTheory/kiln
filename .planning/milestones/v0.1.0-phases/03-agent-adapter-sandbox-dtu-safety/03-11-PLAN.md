# Phase 3 D-143 invariant 7: secrets_presence_map_non_empty
defp check_secrets_presence_map_non_empty! do
  env = Application.get_env(:kiln, :env, :prod)

  {present, missing} =
    [:anthropic_api_key, :openai_api_key, :google_api_key, :ollama_host]
    |> Enum.split_with(&Kiln.Secrets.present?/1)

  Logger.info(
    "provider_keys_loaded=\#{inspect(present)} provider_keys_missing=\#{inspect(missing)}"
  )

  case {env, present} do
    {:prod, []} ->
      raise Error,
        invariant: :secrets_presence_map_non_empty,
        details: %{present: present, missing: missing, env: env},
        remediation_hint:
          "At least one provider API key must be present in :prod. " <>
            "Set ANTHROPIC_API_KEY (or another provider) before booting."

    _ ->
      :ok
  end
end

# Phase 3 D-143 invariant 8: no_prior_boot_sandbox_orphans
defp check_no_prior_boot_sandbox_orphans! do
  current_epoch = Kiln.Sandboxes.OrphanSweeper.boot_epoch_now()

  case Kiln.Sandboxes.DockerDriver.list_orphans(current_epoch) do
    [] ->
      :ok

    ids when is_list(ids) ->
      # Sweeping IS the cure; emit audit + force-rm. NEVER fatal on
      # non-empty (only fatal if docker CLI unreachable).
      for id <- ids do
        Logger.info("Boot sweep: removing orphan container #{id}")
        _ = System.cmd("docker", ["rm", "-f", id], stderr_to_stdout: true)
      end

      :ok
  end
rescue
  e ->
    raise Error,
      invariant: :no_prior_boot_sandbox_orphans,
      details: %{exception: inspect(e)},
      remediation_hint:
        "docker CLI unreachable. Ensure Docker daemon is running before booting Kiln."
end

```elixir
def start_run(run_id) when is_binary(run_id) do
  run = Runs.get!(run_id)

  # D-131 — typed missing-key gate BEFORE any LLM call.
  required_providers = required_providers_for_run(run)

  case missing_providers(required_providers) do
    [] ->
      # Existing start logic continues:
      do_start(run)

    missing ->
      # Raise typed block reason; caller (HTTP endpoint or Mix task) can
      # render the playbook + transition the run to :blocked.
      raise Kiln.Blockers.BlockedError,
        reason: :missing_api_key,
        run_id: run_id,
        context: %{
          missing_providers: missing,
          required_providers: required_providers
        }
  end
end

# D-131 derive providers from run's preset's role->model mapping.
defp required_providers_for_run(run) do
  preset =
    case run.model_profile_snapshot do
      %{"preset" => p} when is_binary(p) -> String.to_existing_atom(p)
      _ -> :elixir_lib
    end

  Kiln.ModelRegistry.resolve(preset)
  |> Map.values()
  |> Enum.map(&provider_for_model(&1.model))
  |> Enum.uniq()
end

defp provider_for_model("claude-" <> _), do: :anthropic_api_key
defp provider_for_model("gpt-" <> _), do: :openai_api_key
defp provider_for_model("gemini-" <> _), do: :google_api_key
defp provider_for_model(_), do: :ollama_host

defp missing_providers(required) do
  Enum.reject(required, &Kiln.Secrets.present?/1)
end
```
