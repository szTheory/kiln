```elixir
defmodule Kiln.SandboxCase do
  @moduledoc """
  Shared setup for sandbox integration tests. Gated by `@tag :docker`
  so CI workers without a Docker daemon skip these tests.

  Tracks container IDs created during the test and `docker rm -f`'s them
  on exit regardless of test outcome (defense-in-depth against OrphanSweeper).
  """

  use ExUnit.CaseTemplate

  using do
    quote do
      @moduletag :docker
      import Kiln.DockerHelper
    end
  end

  setup tags do
    if tags[:docker] && System.find_executable("docker") == nil do
      {:skip, "docker CLI not available"}
    else
      container_ids = :ets.new(:sandbox_case_containers, [:public, :bag])

      on_exit(fn ->
        for {_, id} <- :ets.tab2list(container_ids) do
          System.cmd("docker", ["rm", "-f", id], stderr_to_stdout: true)
        end
      end)

      {:ok, container_ids: container_ids}
    end
  end
end
```

```elixir
defmodule Kiln.DockerHelper do
  @moduledoc """
  Helpers for live-docker tests. Callers should use via `Kiln.SandboxCase`.
  """

  @spec track_container(:ets.tid(), String.t()) :: true
  def track_container(ets_tid, id), do: :ets.insert(ets_tid, {:container, id})

  @spec docker_available? :: boolean()
  def docker_available?, do: System.find_executable("docker") != nil

  @spec exec_in_container(String.t(), [String.t()]) :: {Collectable.t(), non_neg_integer()}
  def exec_in_container(container_id, args) do
    System.cmd("docker", ["exec", container_id | args], stderr_to_stdout: true)
  end
end
```
