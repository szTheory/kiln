defmodule Kiln.Repo.Migrations.ExtendAuditEventKindsP3 do
  use Ecto.Migration

  # Phase 3 D-145 / D-106 extension: adds 9 new event kinds.

  #   :dtu_health_degraded, :factory_circuit_opened, :factory_circuit_closed,

  # Kiln.Audit.EventKind.values_as_strings/0 — never enumerate atoms
  # in the migration body (Plan 02-01 decision (a) — drift-resistant up/0).
  #
  # down/0 HARDCODES the prior 25-atom list (Plan 02-01 decision (d) —
  # reading EventKind at rollback time would observe the post-migration
  # 34-atom source and silently no-op).

  @prior_p2_kinds_25 ~w(
    block_raised block_resolved escalation_triggered
    stage_input_rejected artifact_written integrity_violation
    workflow_loaded workflow_loaded_from_cache stage_started stage_completed
    stage_failed stage_skipped run_started run_transitioned run_completed
    run_blocked run_escalated external_op_intent_recorded external_op_completed
    external_op_failed audit_replay_started gc_swept_artifact integrity_check_passed
    cas_put cas_get
  )

  def up do
    list = Kiln.Audit.EventKind.values_as_strings()
    formatted = Enum.map_join(list, ", ", &"'#{&1}'")

    execute(
      "ALTER TABLE audit_events DROP CONSTRAINT IF EXISTS audit_events_event_kind_check",
      "ALTER TABLE audit_events DROP CONSTRAINT IF EXISTS audit_events_event_kind_check"
    )

    execute(
      "ALTER TABLE audit_events ADD CONSTRAINT audit_events_event_kind_check CHECK (event_kind IN (#{formatted}))",

      "ALTER TABLE audit_events ADD CONSTRAINT audit_events_event_kind_check CHECK (event_kind IN (#{formatted}))",
