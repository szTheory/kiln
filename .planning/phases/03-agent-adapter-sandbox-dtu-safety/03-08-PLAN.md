defmodule Kiln.Sandboxes.Driver do
  @moduledoc """
  Behaviour for sandbox drivers (D-115). P3 ships exactly one impl:
  `Kiln.Sandboxes.DockerDriver` (MuonTrap-wrapped `docker run`).

  Phase 9 hardening may add Kata/gVisor/Firecracker drivers behind this
  behaviour with zero call-site changes (D-118 — REJECTED for P3).
  """

  alias Kiln.Sandboxes.ContainerSpec

  @type run_result :: %{
          required(:exit_code) => integer(),
          required(:container_id) => String.t(),
          optional(:oom_killed) => boolean(),
          optional(:started_at) => String.t(),
          optional(:finished_at) => String.t()
        }

  @callback run_stage(ContainerSpec.t()) :: {:ok, run_result()} | {:error, term()}
  @callback kill(container_id :: String.t()) :: :ok | {:error, term()}
  @callback list_orphans(boot_epoch :: integer()) :: [String.t()]
end

defmodule Kiln.Sandboxes.DockerDriver do
  @moduledoc """
  Live Docker driver (D-115). Wraps `docker run` via `MuonTrap.cmd/3`
  for crash-safe subprocess-tree kill (cgroup-v2 on Linux; port-close
  semantic on macOS).

  Argv assembly per D-117 hardened option set (full list — see
  `assemble_argv/1`). Operator `docker run` repro: every telemetry
  `:start` event carries the EXACT argv as metadata.

  External operation: every `run_stage/1` writes `external_operations`
  two-phase intent (`op_kind: "docker_run"`).
  """

  @behaviour Kiln.Sandboxes.Driver

  require Logger
  alias Kiln.Sandboxes.ContainerSpec
  alias Kiln.ExternalOperations

  @impl true
  def run_stage(%ContainerSpec{} = spec) do
    correlation_id = Logger.metadata()[:correlation_id] || Ecto.UUID.generate()
    run_id = spec.labels["kiln.run_id"]
    stage_run_id = spec.labels["kiln.stage_run_id"]
    idempotency_key = "run:\#{run_id}:stage:\#{stage_run_id}:docker_run:\#{System.system_time(:millisecond)}"

    {_status, op} =
      ExternalOperations.fetch_or_record_intent(idempotency_key, %{
        op_kind: "docker_run",
        intent_payload: %{"image" => spec.image_ref, "labels" => spec.labels},
        run_id: run_id,
        stage_id: stage_run_id
      })

    argv = assemble_argv(spec)

    meta = %{
      run_id: run_id,
      stage_id: stage_run_id,
      docker_argv: argv,
      image_ref: spec.image_ref,
      image_digest: spec.image_digest
    }

    :telemetry.span([:kiln, :sandbox, :docker, :run], meta, fn ->
      case do_run(argv) do
        {:ok, container_id, exit_code} ->
          inspect_result = inspect_container(container_id)

          ExternalOperations.complete_op(op,
            Map.merge(%{"container_id" => container_id, "exit_code" => exit_code}, inspect_result)
          )

          {{:ok, Map.merge(%{exit_code: exit_code, container_id: container_id}, atomize_inspect(inspect_result))}, meta}

        {:error, reason} = err ->
          ExternalOperations.fail_op(op, %{"reason" => inspect(reason)})
          {err, meta}
      end
    end)
  end

  @impl true
  def kill(container_id) when is_binary(container_id) do
    case System.cmd("docker", ["rm", "-f", container_id], stderr_to_stdout: true) do
      {_, 0} -> :ok
      {err, code} -> {:error, {code, err}}
    end
  end

  @impl true
  def list_orphans(boot_epoch) when is_integer(boot_epoch) do
    # ex_docker_engine_api LIST; D-120
    # Filter by label kiln.boot_epoch != boot_epoch.
    # P3 stub: shell out to docker ps; Phase 9 hardening migrates to API client.
    {out, 0} = System.cmd("docker",
      ["ps", "-a", "--filter", "label=kiln.run_id", "--format", "{{.ID}} {{.Label \"kiln.boot_epoch\"}}"]
    )

    out
    |> String.split("\n", trim: true)
    |> Enum.flat_map(fn line ->
      case String.split(line, " ", parts: 2) do
        [id, epoch_str] ->
          case Integer.parse(epoch_str) do
            {n, _} when n != boot_epoch -> [id]
            _ -> []
          end

        _ -> []
      end
    end)
  end

  # ---- argv assembly ---------------------------------------------------

  @doc false
  def __assemble_argv__(spec), do: assemble_argv(spec)

  defp assemble_argv(%ContainerSpec{} = spec) do
    base = ["run", "--rm"]
    network = ["--network", spec.network]

    cap = if spec.cap_drop_all, do: ["--cap-drop=ALL"], else: []
    sec_opts = Enum.flat_map(spec.security_opts, fn opt -> ["--security-opt=" <> opt] end)
    read_only = if spec.read_only, do: ["--read-only"], else: []

    tmpfs = Enum.flat_map(spec.tmpfs_mounts, fn {path, size} -> ["--tmpfs", "\#{path}:rw,nosuid,size=\#{size}"] end)

    user = ["--user", spec.user]

    limits =
      [
        ["--memory=" <> to_string(spec.limits["memory"])],
        ["--memory-swap=" <> to_string(spec.limits["memory_swap"])],
        ["--cpus=" <> to_string(spec.limits["cpus"])],
        ["--pids-limit=" <> to_string(spec.limits["pids_limit"])],
        ["--ulimit", "nofile=" <> to_string(spec.limits["ulimit_nofile"])],
        ["--ulimit", "nproc=" <> to_string(spec.limits["ulimit_nproc"])]
      ]
      |> List.flatten()

    stop_timeout = ["--stop-timeout", to_string(spec.stop_timeout)]

    labels = Enum.flat_map(spec.labels, fn {k, v} -> ["--label", "\#{k}=\#{v}"] end)

    env_file = if spec.env_file_path, do: ["--env-file", spec.env_file_path], else: []

    workdir = ["--workdir", spec.workdir]

    init = if spec.init, do: ["--init"], else: []

    dns = Enum.flat_map(spec.dns, fn ip -> ["--dns", ip] end)

    extra_hosts = Enum.flat_map(spec.extra_hosts, fn h -> ["--add-host", h] end)

    ipv6 =
      if spec.ipv6_disabled, do: ["--sysctl", "net.ipv6.conf.all.disable_ipv6=1"], else: []

    image = [spec.image_digest || spec.image_ref]

    cmd = spec.cmd || []

    base ++ network ++ cap ++ sec_opts ++ read_only ++ tmpfs ++ user ++ limits ++
      stop_timeout ++ labels ++ env_file ++ workdir ++ init ++ dns ++ extra_hosts ++ ipv6 ++
      image ++ cmd
  end

  # ---- subprocess wrap -------------------------------------------------

  defp do_run(argv) do
    case MuonTrap.cmd("docker", argv, stderr_to_stdout: true) do
      {output, 0} ->
        # The container ID is on the LAST non-empty line of stdout when run
        # without -d (we DO use --rm + foreground so output is the program's
        # stdout). For run-with-detach mode we'd parse differently; P3 uses
        # foreground per simplicity.
        # Generate a synthetic container_id from labels for ID-tracking:
        # In foreground mode docker prints stage stdout; container_id obtained
        # via docker ps lookup against labels.
        container_id = lookup_container_by_labels(output)
        {:ok, container_id, 0}

      {output, exit_code} ->
        container_id = lookup_container_by_labels(output)
        {:ok, container_id, exit_code}
    end
  rescue
    e -> {:error, {:exception, e}}
  end

  # P3 simplification: container_id captured via parallel `docker ps --filter label=...`
  # at end of run. For tests we accept "" and treat as best-effort.
  defp lookup_container_by_labels(_output) do
    ""
  end

  defp inspect_container("") do
    %{"oom_killed" => false, "exit_code_inspect" => -1}
  end

  defp inspect_container(id) when is_binary(id) do
    case System.cmd("docker", ["inspect", "--format", "{{.State.OOMKilled}}|{{.State.ExitCode}}|{{.State.StartedAt}}|{{.State.FinishedAt}}", id], stderr_to_stdout: true) do
      {out, 0} ->
        [oom, exit_code, started, finished] = String.split(String.trim(out), "|", parts: 4)
        %{
          "oom_killed" => oom == "true",
          "exit_code_inspect" => String.to_integer(exit_code),
          "started_at" => started,
          "finished_at" => finished
        }

      _ -> %{}
    end
  end

  defp atomize_inspect(map) do
    %{
      oom_killed: map["oom_killed"] || false,
      started_at: map["started_at"],
      finished_at: map["finished_at"]
    }
  end
end

defmodule Kiln.Sandboxes.OrphanSweeper do
  @moduledoc """
  Boot-time + periodic sweep of orphaned sandbox containers (D-120 /
  SAND-01 cleanup). Runs ONCE at boot — `Kiln.BootChecks` 8th invariant
  (Wave 5 wires this) — and again every `@periodic_scan_ms`
  (defense-in-depth for long-running hosts).

  Strategy:
    1. Enumerate `docker ps -a --filter label=kiln.run_id`.
    2. Filter where `kiln.boot_epoch` label != current boot epoch.
    3. For each survivor: emit `orphan_container_swept` audit event
       + `docker rm -f <id>`.

  P3 implementation uses `System.cmd("docker", ...)` for both LIST and
  DESTROY; Phase 9 hardening may migrate LIST to `ex_docker_engine_api`
  for richer typed responses (D-120 trade — this implementation works
  without ex_docker_engine_api dep at first call site).
  """

  use GenServer
  require Logger
  alias Kiln.{Audit, Sandboxes.DockerDriver}

  @periodic_scan_ms :timer.minutes(15)

  @spec start_link(keyword()) :: GenServer.on_start()
  def start_link(opts), do: GenServer.start_link(__MODULE__, opts, name: __MODULE__)

  @spec sweep_now! :: :ok
  def sweep_now! do
    GenServer.call(__MODULE__, :scan_synchronous, 30_000)
  end

  @impl true
  def init(_opts) do
    boot_epoch = boot_epoch_now()
    state = %{boot_epoch: boot_epoch}
    send(self(), :boot_scan)
    {:ok, state}
  end

  @impl true
  def handle_info(:boot_scan, state) do
    state = do_scan(state)
    Process.send_after(self(), :periodic_scan, @periodic_scan_ms)
    {:noreply, state}
  end

  @impl true
  def handle_info(:periodic_scan, state) do
    state = do_scan(state)
    Process.send_after(self(), :periodic_scan, @periodic_scan_ms)
    {:noreply, state}
  end

  @impl true
  def handle_info(_msg, state), do: {:noreply, state}

  @impl true
  def handle_call(:scan_synchronous, _from, state) do
    state = do_scan(state)
    {:reply, :ok, state}
  end

  defp do_scan(%{boot_epoch: boot_epoch} = state) do
    case DockerDriver.list_orphans(boot_epoch) do
      [] ->
        state

      ids when is_list(ids) ->
        for id <- ids do
          age_seconds = 0  # P3: not computing age; Phase 9 hardening adds via inspect

          _ = Audit.append(%{
            event_kind: :orphan_container_swept,
            run_id: nil,
            stage_id: nil,
            correlation_id: Logger.metadata()[:correlation_id] || Ecto.UUID.generate(),
            payload: %{
              "container_id" => id,
              "boot_epoch_found" => 0,
              "age_seconds" => age_seconds
            }
          })

          {_, _} = System.cmd("docker", ["rm", "-f", id], stderr_to_stdout: true)
        end

        state
    end
  rescue
    e ->
      Logger.error("OrphanSweeper scan failed: #{inspect(e)}")
      state
  end

  @doc "Current BEAM boot epoch (monotonic ms — used in container labels)."
  @spec boot_epoch_now :: integer()
  def boot_epoch_now, do: System.monotonic_time(:millisecond)
end
