defmodule Kiln.Pricing do
  @moduledoc """
  Pricing-table lookup (D-146). Loads `priv/pricing/v1/<provider>.exs`
  files at compile time and exposes `estimate_usd/3`.

  On unknown model: returns `Decimal.new(0)` + emits a `Logger.warning/1`
  (graceful — pricing-table drift should not crash a run; `mix kiln.pricing.check`
  surfaces drift in CI).

  See also `mix kiln.pricing.check` (Wave 5/6 ships the WARN-only drift task).
  """

  require Logger

  @pricing_dir Path.expand("../../priv/pricing/v1", __DIR__)
  @providers [:anthropic, :openai, :google, :ollama]

  @pricing (for provider <- @providers, into: %{} do
              path = Path.join(@pricing_dir, "\#{provider}.exs")
              @external_resource path

              data =
                case File.read(path) do
                  {:ok, _} -> Code.eval_file(path) |> elem(0)
                  {:error, _} -> %{}
                end

              for {model, rates} <- data, into: %{} do
                {model, rates}
              end
              |> then(&{provider, &1})
            end)

  @flat (for {_provider, models} <- @pricing, into: %{} do
           models
         end)

  @spec estimate_usd(String.t() | nil, non_neg_integer(), non_neg_integer()) :: Decimal.t()
  def estimate_usd(nil, _in, _out), do: Decimal.new(0)

  def estimate_usd(model, input_tokens, output_tokens)
      when is_binary(model) and is_integer(input_tokens) and is_integer(output_tokens) do
    case Map.get(@flat, model) do
      nil ->
        Logger.warning("Kiln.Pricing.estimate_usd: unknown model #{model} — returning 0")
        Decimal.new(0)

      %{input_per_mtok_usd: i, output_per_mtok_usd: o} ->
        in_cost = Decimal.mult(Decimal.div(Decimal.new(input_tokens), Decimal.new(1_000_000)), i)
        out_cost = Decimal.mult(Decimal.div(Decimal.new(output_tokens), Decimal.new(1_000_000)), o)
        Decimal.add(in_cost, out_cost)
    end
  end

  @spec known_model?(String.t()) :: boolean()
  def known_model?(model) when is_binary(model), do: Map.has_key?(@flat, model)

  @spec all_known_models :: [String.t()]
  def all_known_models, do: Map.keys(@flat)
end
```

defmodule Kiln.ModelRegistry do
  @moduledoc """
  Preset → role → model resolution + fallback chain (D-105..D-108 / OPS-03).

  Loads `priv/model_registry/<preset>.exs` files at compile time. P3 ships
  `:same_provider` fallback policy on every role; Phase 5+ flips OpenAI live
  and operator may switch to `:cross_provider`.

  Public API:
    * `resolve/2` — preset name + per-stage overrides → role->spec map
    * `next/2` — given current model + fault class, return next model in chain
    * `adapter_for/1` — model id → adapter module (provider routing)
  """

  require Logger
  alias Kiln.ModelRegistry.Preset

  @presets ~w(elixir_lib phoenix_saas_feature typescript_web_feature python_cli bugfix_critical docs_update)a
  @presets_dir Path.expand("../../priv/model_registry", __DIR__)

  @presets_data (for preset <- @presets, into: %{} do
                   path = Path.join(@presets_dir, "\#{preset}.exs")
                   @external_resource path

                   case File.read(path) do
                     {:ok, _} ->
                       roles = Code.eval_file(path) |> elem(0)
                       {preset, %Preset{name: preset, roles: roles}}

                     {:error, :enoent} ->
                       raise CompileError,
                         description:
                           "Missing preset file priv/model_registry/\#{preset}.exs — every D-57 preset MUST have a definition file"
                   end
                 end)

  @spec all_presets :: [atom()]
  def all_presets, do: @presets

  @spec resolve(atom(), map()) :: %{atom() => map()}
  def resolve(preset_name, stage_overrides \\ %{}) when is_atom(preset_name) and is_map(stage_overrides) do
    preset = Map.fetch!(@presets_data, preset_name)
    Map.merge(preset.roles, stage_overrides)
  end

  @spec next(String.t(), atom(), atom()) :: {:ok, String.t()} | {:exhausted, String.t()}
  def next(current_model, _fault_class, preset_name)
      when is_binary(current_model) and is_atom(preset_name) do
    preset = Map.fetch!(@presets_data, preset_name)

    chain =
      preset.roles
      |> Map.values()
      |> Enum.flat_map(fn %{model: m, fallback: f} -> [m | f] end)
      |> Enum.uniq()

    case Enum.drop_while(chain, &(&1 != current_model)) do
      [_current, next | _] -> {:ok, next}
      _ -> {:exhausted, current_model}
    end
  end

  @spec adapter_for(String.t()) :: module()
  def adapter_for("claude-" <> _), do: Kiln.Agents.Adapter.Anthropic
  def adapter_for("gpt-" <> _), do: Kiln.Agents.Adapter.OpenAI
  def adapter_for("gemini-" <> _), do: Kiln.Agents.Adapter.Google
  def adapter_for(_), do: Kiln.Agents.Adapter.Ollama
end
```

defmodule Mix.Tasks.Kiln.Registry.Show do
  use Mix.Task
  @shortdoc "Print resolved role->model mapping for a preset"

  @impl Mix.Task
  def run([preset_str]) do
    Mix.Task.run("app.start")
    preset = String.to_existing_atom(preset_str)

    mapping = Kiln.ModelRegistry.resolve(preset)

    IO.puts("Preset: \#{preset}")

    for {role, spec} <- Enum.sort_by(mapping, &elem(&1, 0)) do
      IO.puts("  \#{role}:")
      IO.puts("    model:           \#{spec.model}")
      IO.puts("    fallback:        \#{inspect(spec.fallback)}")
      IO.puts("    fallback_policy: \#{spec.fallback_policy}")
    end
  end

  def run(_), do: Mix.raise("Usage: mix kiln.registry.show <preset_name>")
end

defmodule Kiln.Agents.BudgetGuard do
  @moduledoc """
  Per-call cost pre-flight (D-138 / AGENT-05). Runs BEFORE every LLM
  call from `Kiln.Stages.StageWorker.perform/1`. Strict — NO
  `KILN_BUDGET_OVERRIDE` escape hatch (per D-138 explicit choice).

  7-step check order (D-138):
    1. Read `runs.caps_snapshot["max_tokens_usd"]`
    2. SUM completed-stage spend for run
    3. Compute `remaining_budget_usd`
    4. Call `adapter.count_tokens/1` (Anthropic free pre-flight; others use estimate)
    5. `Pricing.estimate_usd(model, input, max_output)`
    6. Compare estimated to remaining
    7. Emit `budget_check_passed` OR raise `:budget_exceeded` + emit `budget_check_failed`

  All 7 steps inside ONE telemetry span: `[:kiln, :agents, :budget_guard, :check]`.

  On breach: writes `budget_check_failed` audit, calls
  `Kiln.Notifications.desktop(:budget_exceeded, ...)`, transitions run
  through `:blocked` (RunDirector consumer), and raises `BlockedError`.
  """

  require Logger
  alias Kiln.{Audit, Pricing, Notifications}
  alias Kiln.Agents.{Prompt, Response}

  @spec check!(Prompt.t(), keyword()) :: :ok | no_return()
  def check!(%Prompt{} = prompt, opts) do
    run_id = Keyword.fetch!(opts, :run_id)
    stage_id = Keyword.fetch!(opts, :stage_id)
    model = prompt.model || Keyword.fetch!(opts, :model)
    adapter = Keyword.fetch!(opts, :adapter)

    meta = %{run_id: run_id, stage_id: stage_id, model: model}

    :telemetry.span([:kiln, :agents, :budget_guard, :check], meta, fn ->
      result = do_check(prompt, run_id, stage_id, model, adapter)
      {result, meta}
    end)
  end

  defp do_check(prompt, run_id, stage_id, model, adapter) do
    # Step 1 — read caps_snapshot
    run = Kiln.Runs.get!(run_id)
    cap = decimalize(get_in(run.caps_snapshot, ["max_tokens_usd"]) || "0")

    # Step 2 — sum completed-stage spend
    spent = sum_stage_spend(run_id)

    # Step 3 — compute remaining
    remaining = Decimal.sub(cap, spent)

    # Step 4 — pre-flight token count via adapter
    {:ok, tokens_in} = adapter.count_tokens(prompt)

    # Step 5 — estimate USD
    estimated = Pricing.estimate_usd(model, tokens_in, prompt.max_tokens)

    # Step 6 — compare
    breach? = Decimal.compare(estimated, remaining) == :gt

    correlation_id = Logger.metadata()[:correlation_id] || Ecto.UUID.generate()

    if breach? do
      # Step 7a — emit failed audit + notify + raise
      _ = Audit.append(%{
        event_kind: :budget_check_failed,
        run_id: run_id,
        stage_id: stage_id,
        correlation_id: correlation_id,
        payload: %{
          "estimated_usd" => Decimal.to_string(estimated),
          "remaining_usd" => Decimal.to_string(remaining),
          "cap_usd" => Decimal.to_string(cap),
          "spent_usd" => Decimal.to_string(spent),
          "model" => model
        }
      })

      _ = Notifications.desktop(:budget_exceeded, %{
        run_id: run_id,
        estimated_usd: Decimal.to_string(estimated),
        remaining_usd: Decimal.to_string(remaining),
        severity: "halt"
      })

      raise Kiln.Blockers.BlockedError,
        reason: :budget_exceeded,
        run_id: run_id,
        context: %{
          estimated_usd: Decimal.to_string(estimated),
          remaining_usd: Decimal.to_string(remaining),
          model: model
        }
    else
      # Step 7b — emit passed audit
      _ = Audit.append(%{
        event_kind: :budget_check_passed,
        run_id: run_id,
        stage_id: stage_id,
        correlation_id: correlation_id,
        payload: %{
          "estimated_usd" => Decimal.to_string(estimated),
          "remaining_usd" => Decimal.to_string(remaining),
          "model" => model
        }
      })

      :ok
    end
  end

  defp sum_stage_spend(run_id) do
    import Ecto.Query

    Kiln.Repo.one(
      from sr in Kiln.Stages.StageRun,
        where: sr.run_id == ^run_id and sr.state in [^:completed, ^:failed],
        select: sum(sr.tokens_used_usd)
    )
    |> case do
      nil -> Decimal.new(0)
      d -> d
    end
  end

  defp decimalize(%Decimal{} = d), do: d
  defp decimalize(s) when is_binary(s), do: Decimal.new(s)
  defp decimalize(n) when is_integer(n), do: Decimal.new(n)
  defp decimalize(n) when is_float(n), do: Decimal.from_float(n)
end

defmodule Kiln.Agents.TelemetryHandler do
  @moduledoc """
  Attaches at boot (Wave 5 wires Application.start) to consume
  `[:kiln, :agent, :call, *]` events. Writes ONE `model_routing_fallback`
  audit event per LLM call when `actual_model_used != requested_model`
  (OPS-02 / D-106 visibility — silent fallback impossible).

  Mirrors `Kiln.Telemetry.ObanHandler.attach/0` shape (P1 D-46 pattern).
  """

  @handler_id {__MODULE__, :agent_call_lifecycle}

  @spec attach :: :ok | {:error, :already_exists}
  def attach do
    :telemetry.attach_many(
      @handler_id,
      [
        [:kiln, :agent, :call, :start],
        [:kiln, :agent, :call, :stop],
        [:kiln, :agent, :call, :exception]
      ],
      &__MODULE__.handle_event/4,
      nil
    )
  end

  @spec detach :: :ok
  def detach, do: :telemetry.detach(@handler_id)

  @spec handle_event(list(), map(), map(), term()) :: :ok
  def handle_event([:kiln, :agent, :call, :stop], measurements, metadata, _config) do
    requested = metadata[:requested_model]
    actual = metadata[:actual_model_used]

    if requested && actual && requested != actual do
      Kiln.Audit.append(%{
        event_kind: :model_routing_fallback,
        run_id: metadata[:run_id],
        stage_id: metadata[:stage_id],
        correlation_id: Logger.metadata()[:correlation_id] || Ecto.UUID.generate(),
        payload: %{
          "requested_model" => to_string(requested),
          "actual_model_used" => to_string(actual),
          "fallback_reason" => "fallback_in_adapter",
          "tier_crossed" => tier_crossed?(requested, actual),
          "attempt_number" => metadata[:attempt_number] || 1,
          "provider_http_status" => metadata[:provider_http_status],
          "provider" => to_string(metadata[:provider] || "unknown"),
          "role" => to_string(metadata[:role] || "unknown"),
          "wall_clock_ms" => System.convert_time_unit(measurements[:duration] || 0, :native, :millisecond)
        }
      })
    end

    :ok
  end

  def handle_event(_event, _measurements, _metadata, _config), do: :ok

  defp tier_crossed?(requested, actual) when is_binary(requested) and is_binary(actual) do
    requested_tier = tier_of(requested)
    actual_tier = tier_of(actual)
    requested_tier != actual_tier
  end

  defp tier_crossed?(_, _), do: false

  defp tier_of("claude-opus" <> _), do: :opus
  defp tier_of("claude-sonnet" <> _), do: :sonnet
  defp tier_of("claude-haiku" <> _), do: :haiku
  defp tier_of("gpt-4o" <> _), do: :flagship
  defp tier_of("gpt-4o-mini" <> _), do: :mini
  defp tier_of(_), do: :unknown
end
