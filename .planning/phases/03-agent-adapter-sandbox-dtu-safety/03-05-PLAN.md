```elixir
defmodule Kiln.Agents.Prompt do
  @moduledoc """
  Provider-agnostic LLM prompt envelope. Adapters translate to their
  native shape inside `complete/2`/`stream/2`.

  Minimum-viable shape — Phase 4 may extend with role-specific tool-call
  metadata. P3 keeps the surface narrow per Claude's-discretion policy
  in 03-CONTEXT.md (only fields explicit consumers need).

  Note: `:metadata` is the carrier for `kiln_ctx` (Logger metadata pack)
  passed across Oban boundaries — ALSO the carrier for `%Secrets.Ref{}`
  values that resolve to api_key inside the adapter HTTP-call frame.
  Hence `:metadata` is excluded from `@derive Jason.Encoder` (D-133 Layer 1).
  """

  @derive {Jason.Encoder,
           only: [:messages, :system, :model, :max_tokens, :temperature, :tools]}

  defstruct [
    :model,
    system: nil,
    messages: [],
    max_tokens: 4096,
    temperature: 1.0,
    tools: [],
    metadata: %{}
  ]

  @type message :: %{required(:role) => :user | :assistant, required(:content) => term()}
  @type tool :: map()

  @type t :: %__MODULE__{
          model: String.t() | nil,
          system: String.t() | nil,
          messages: [message()],
          max_tokens: pos_integer(),
          temperature: float(),
          tools: [tool()],
          metadata: map()
        }
end
```

```elixir
defmodule Kiln.Agents.Adapter.Anthropic do
  @moduledoc """
  LIVE Anthropic adapter (D-101). Wraps Anthropix 0.6.2 for the message
  surface; uses raw `Req` for `count_tokens` (Anthropix 0.6 does not
  expose this — see PATTERNS.md anti-pattern #3).

  Public surface: implements `Kiln.Agents.Adapter`. All HTTP traffic
  goes through the named `Kiln.Finch` pool routed to the Anthropic host
  (D-109). Authorization header is built inside `call_http/2` via
  `Kiln.Secrets.reveal!/1` — the SOLE raw-string boundary in this file
  (D-131..D-133 Layer 1; grep audit pins call site count to 1).

  Telemetry contract (D-110):
    * `[:kiln, :agent, :call, :start]` — measurements: `system_time`;
      metadata: `provider, role, run_id, stage_id, requested_model`
    * `[:kiln, :agent, :call, :stop]` — measurements:
      `duration_native, tokens_in, tokens_out, cost_usd`;
      metadata: `provider, role, run_id, stage_id, requested_model,
                 actual_model_used, fallback?`
    * `[:kiln, :agent, :call, :exception]` — measurements:
      `duration_native`; metadata: `kind, reason, stacktrace, ...`

  ExternalOperations contract: every `complete/2` and `stream/2` call
  records a two-phase intent (`op_kind: "llm_complete"` or `"llm_stream"`,
  idempotency_key per attempt). On success, `complete_op` writes the
  result; on error, `fail_op`.
  """

  @behaviour Kiln.Agents.Adapter

  alias Kiln.Agents.{Prompt, Response}
  alias Kiln.{ExternalOperations, Pricing, Secrets, Telemetry}

  @anthropic_base_url "https://api.anthropic.com"
  @anthropic_version "2023-06-01"

  @impl true
  def capabilities do
    %{
      streaming: true,
      tools: true,
      thinking: true,
      vision: true,
      json_schema_mode: true
    }
  end

  @impl true
  def complete(%Prompt{} = prompt, opts) do
    ctx = Keyword.get(opts, :kiln_ctx, %{})
    if map_size(ctx) > 0, do: Telemetry.unpack_ctx(ctx)

    run_id = Logger.metadata()[:run_id]
    stage_id = Logger.metadata()[:stage_id]
    requested_model = prompt.model || Keyword.fetch!(opts, :model)
    role = Keyword.get(opts, :role)
    attempt = Keyword.get(opts, :attempt, 1)

    idempotency_key = "run:\#{run_id}:stage:\#{stage_id}:llm_complete:\#{attempt}"

    {_status, op} =
      ExternalOperations.fetch_or_record_intent(idempotency_key, %{
        op_kind: "llm_complete",
        intent_payload: prompt_audit_shape(prompt),
        run_id: run_id,
        stage_id: stage_id
      })

    meta_start = %{
      run_id: run_id,
      stage_id: stage_id,
      requested_model: requested_model,
      provider: :anthropic,
      role: role
    }

    :telemetry.span([:kiln, :agent, :call], meta_start, fn ->
      case do_complete(prompt, opts) do
        {:ok, %Response{} = response} ->
          ExternalOperations.complete_op(op, %{
            "result" => "ok",
            "tokens_in" => response.tokens_in,
            "tokens_out" => response.tokens_out,
            "actual_model_used" => response.actual_model_used,
            "cost_usd" => Decimal.to_string(response.cost_usd || Decimal.new(0))
          })

          measurements = %{
            tokens_in: response.tokens_in,
            tokens_out: response.tokens_out,
            cost_usd: response.cost_usd
          }

          extra_meta = %{
            actual_model_used: response.actual_model_used,
            fallback?: response.actual_model_used != requested_model
          }

          {{:ok, response}, Map.merge(measurements, Map.merge(meta_start, extra_meta))}

        {:error, reason} = err ->
          ExternalOperations.fail_op(op, %{"reason" => inspect(reason)})
          {err, meta_start}
      end
    end)
  end

  @impl true
  def stream(%Prompt{} = prompt, opts) do
    # D-103: stream/2 returns Enumerable passthrough; NO PubSub in P3.
    # Telemetry per chunk via Stream.each.
    ctx = Keyword.get(opts, :kiln_ctx, %{})
    if map_size(ctx) > 0, do: Telemetry.unpack_ctx(ctx)

    run_id = Logger.metadata()[:run_id]
    stage_id = Logger.metadata()[:stage_id]
    requested_model = prompt.model || Keyword.fetch!(opts, :model)
    start_time = System.monotonic_time()

    case do_stream(prompt, opts) do
      {:ok, raw_stream} ->
        wrapped =
          Stream.each(raw_stream, fn chunk ->
            :telemetry.execute(
              [:kiln, :agent, :stream, :chunk],
              %{byte_size: chunk_byte_size(chunk), elapsed_since_start: System.monotonic_time() - start_time},
              %{run_id: run_id, stage_id: stage_id, actual_model_used: requested_model}
            )
          end)

        {:ok, wrapped}

      {:error, _} = err ->
        err
    end
  end

  @impl true
  def count_tokens(%Prompt{} = prompt) do
    # PATTERNS anti-pattern #3: Anthropix 0.6.2 does NOT wrap count_tokens.
    # Build a direct Req call against POST /v1/messages/count_tokens.
    headers = build_headers()

    body = %{
      "model" => prompt.model,
      "messages" => prompt.messages,
      "system" => prompt.system
    }

    case Req.post("\#{@anthropic_base_url}/v1/messages/count_tokens",
           finch: Kiln.Finch,
           headers: headers,
           json: body
         ) do
      {:ok, %{status: 200, body: %{"input_tokens" => n}}} ->
        {:ok, n}

      {:ok, %{status: status, body: body}} ->
        {:error, {:http_status, status, body}}

      {:error, reason} ->
        {:error, reason}
    end
  end

  # ---- Private helpers -------------------------------------------------

  defp do_complete(%Prompt{} = prompt, opts) do
    headers = build_headers()
    base_url = Keyword.get(opts, :base_url, @anthropic_base_url)

    body = %{
      "model" => prompt.model,
      "messages" => prompt.messages,
      "max_tokens" => prompt.max_tokens
    }

    body = if prompt.system, do: Map.put(body, "system", prompt.system), else: body
    body = if prompt.temperature != 1.0, do: Map.put(body, "temperature", prompt.temperature), else: body
    body = if prompt.tools != [], do: Map.put(body, "tools", prompt.tools), else: body

    case Req.post("\#{base_url}/v1/messages", finch: Kiln.Finch, headers: headers, json: body) do
      {:ok, %{status: 200, body: raw}} ->
        {:ok, build_response(raw)}

      {:ok, %{status: 429, body: body}} ->
        {:error, {:http_status, 429, body}}

      {:ok, %{status: status, body: body}} when status >= 500 ->
        {:error, {:http_status, status, body}}

      {:ok, %{status: status, body: body}} ->
        {:error, {:http_status, status, body}}

      {:error, reason} ->
        {:error, reason}
    end
  end

  defp do_stream(%Prompt{} = prompt, opts) do
    # P3 stub-friendly: when running against Bypass (or in unit tests),
    # the streaming surface degrades gracefully. Real Anthropix.stream/2
    # wiring shipped via Anthropix.init/1.
    case Anthropix.init(Secrets.reveal!(:anthropic_api_key)) do
      client ->
        opts_keyword = [
          model: prompt.model,
          messages: prompt.messages,
          max_tokens: prompt.max_tokens,
          stream: true
        ] ++ if(prompt.system, do: [system: prompt.system], else: [])

        case Anthropix.chat(client, opts_keyword) do
          {:ok, stream} -> {:ok, stream}
          {:error, _} = err -> err
        end
    end
  rescue
    e -> {:error, {:exception, e}}
  end

  defp build_headers do
    [
      {"authorization", "Bearer " <> Secrets.reveal!(:anthropic_api_key)},
      {"anthropic-version", @anthropic_version},
      {"content-type", "application/json"}
    ]
  end

  defp build_response(raw) when is_map(raw) do
    actual_model = Map.get(raw, "model")
    tokens_in = get_in(raw, ["usage", "input_tokens"]) || 0
    tokens_out = get_in(raw, ["usage", "output_tokens"]) || 0

    %Response{
      content: Map.get(raw, "content"),
      stop_reason: Map.get(raw, "stop_reason") |> stop_reason_atom(),
      tokens_in: tokens_in,
      tokens_out: tokens_out,
      cost_usd: Pricing.estimate_usd(actual_model, tokens_in, tokens_out),
      actual_model_used: actual_model,
      raw: raw
    }
  end

  defp stop_reason_atom(nil), do: nil
  defp stop_reason_atom(s) when is_binary(s), do: String.to_atom(s)

  defp prompt_audit_shape(%Prompt{} = p) do
    %{
      "model" => p.model,
      "max_tokens" => p.max_tokens,
      "messages_count" => length(p.messages),
      "tools_count" => length(p.tools)
    }
  end

  defp chunk_byte_size(chunk) when is_binary(chunk), do: byte_size(chunk)
  defp chunk_byte_size(_), do: 0
end
```

```elixir
defmodule Kiln.Agents.StructuredOutput do
  @moduledoc """
  Provider-agnostic structured-output facade (D-104). Dispatches to
  per-provider native mode based on `Adapter.capabilities().json_schema_mode`.

  Native modes:
    * Anthropic — `tool_use` (current Anthropix); future: `output_config.format.json_schema`
    * OpenAI — `response_format: {type: "json_schema", json_schema: {schema: ..., strict: true}}`
    * Google — `function_calling`
    * Ollama — falls back to prompted-JSON + JSV post-validation + 1 retry

  All paths post-validate response via JSV Draft 2020-12 as
  defense-in-depth.
  """

  alias Kiln.Agents.{Adapter, Prompt}

  @type opts :: [adapter: module(), model: String.t(), prompt: Prompt.t(), schema: map()]

  @spec request(map(), opts()) :: {:ok, term()} | {:error, term()}
  def request(schema, opts) when is_map(schema) and is_list(opts) do
    adapter = Keyword.fetch!(opts, :adapter)
    prompt = Keyword.fetch!(opts, :prompt)

    caps = adapter.capabilities()
    schema_root = JSV.build!(schema, default_meta: "https://json-schema.org/draft/2020-12/schema")

    if caps.json_schema_mode do
      # Native — provider-side enforcement; we still post-validate.
      do_native_request(adapter, prompt, schema, schema_root, opts)
    else
      # Prompted JSON + JSV post-validate + 1 retry.
      do_prompted_request(adapter, prompt, schema, schema_root, opts, _attempt = 1)
    end
  end

  defp do_native_request(adapter, prompt, _schema, schema_root, opts) do
    # Provider-specific native mode is wired inside each adapter's
    # complete/2 when opts[:json_schema] is present. Here we just call
    # complete/2 with the schema attached, then validate.
    opts_with_schema = Keyword.put(opts, :json_schema, _schema = %{})

    with {:ok, response} <- adapter.complete(prompt, opts_with_schema),
         {:ok, parsed} <- parse_json_content(response.content),
         {:ok, validated} <- JSV.validate(parsed, schema_root) do
      {:ok, validated}
    end
  end

  defp do_prompted_request(adapter, prompt, schema, schema_root, opts, attempt) do
    schema_str = Jason.encode!(schema)
    augmented_system = (prompt.system || "") <> "\n\nRespond with JSON matching this schema:\n" <> schema_str

    aug_prompt = %{prompt | system: augmented_system}

    with {:ok, response} <- adapter.complete(aug_prompt, opts),
         {:ok, parsed} <- parse_json_content(response.content),
         {:ok, validated} <- JSV.validate(parsed, schema_root) do
      {:ok, validated}
    else
      {:error, reason} when attempt < 2 ->
        do_prompted_request(adapter, prompt, schema, schema_root, opts, attempt + 1)

      err ->
        err
    end
  end

  defp parse_json_content(content) when is_binary(content) do
    case Jason.decode(content) do
      {:ok, parsed} -> {:ok, parsed}
      {:error, _} = err -> err
    end
  end

  defp parse_json_content(_), do: {:error, :non_string_content}
end
```
