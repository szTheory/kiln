defmodule KilnDtu.Chaos do
  @moduledoc """
  D-125 chaos middleware. P3 ships exactly two modes; others are
  declared but unhandled (return 400 with the closed enum).
  """

  @supported ~w(rate_limit_429 outage_503)

  import Plug.Conn

  def maybe_inject(conn) do
    case get_req_header(conn, "x-dtu-chaos") do
      [] ->
        :pass

      ["rate_limit_429"] ->
        # GitHub-accurate Retry-After header (60s typical for secondary rate limits)
        conn = put_resp_header(conn, "retry-after", "60")
        body = Jason.encode!(%{message: "API rate limit exceeded", documentation_url: "https://docs.github.com/rest/overview/resources-in-the-rest-api#rate-limiting"})
        {:short_circuit, send_resp(conn, 429, body)}

      ["outage_503"] ->
        body = Jason.encode!(%{message: "Service Unavailable"})
        {:short_circuit, send_resp(conn, 503, body)}

      [other] when other not in @supported ->
        body = Jason.encode!(%{error: "unsupported_chaos_mode", received: other, supported: @supported})
        {:short_circuit, send_resp(conn, 400, body)}

      _ ->
        :pass
    end
  end
end

defmodule KilnDtu.Handlers.GitHub.Issues do
  use Plug.Router
  alias KilnDtu.Chaos

  plug :match
  plug :dispatch

  get "/" do
    case Chaos.maybe_inject(conn) do
      {:short_circuit, conn} -> conn
      :pass ->
        body = Jason.encode!([%{number: 1, title: "DTU stub issue", state: "open"}])
        conn |> put_resp_content_type("application/json") |> send_resp(200, body)
    end
  end

  get "/:number" do
    case Chaos.maybe_inject(conn) do
      {:short_circuit, conn} -> conn
      :pass ->
        body = Jason.encode!(%{number: String.to_integer(conn.params["number"]), title: "DTU stub", state: "open"})
        conn |> put_resp_content_type("application/json") |> send_resp(200, body)
    end
  end

  match _ do
    send_resp(conn, 501, Jason.encode!(%{error: "endpoint_not_implemented", endpoint: conn.request_path}))
  end
end

defmodule KilnDtu.RouterTest do
  use ExUnit.Case, async: true
  use Plug.Test
  alias KilnDtu.Router

  @opts Router.init([])

  test "GET /healthz returns 200 with status" do
    conn = conn(:get, "/healthz") |> Router.call(@opts)
    assert conn.status == 200
    assert Jason.decode!(conn.resp_body)["status"] == "ok"
  end

  test "unknown endpoint returns 501 with structured error body" do
    conn = conn(:get, "/totally/unknown") |> Router.call(@opts)
    assert conn.status == 501
    body = Jason.decode!(conn.resp_body)
    assert body["error"] == "endpoint_not_implemented"
    assert body["endpoint"] == "/totally/unknown"
  end

  test "chaos rate_limit_429 returns 429 with Retry-After" do
    conn = conn(:get, "/repos/o/r/issues") |> put_req_header("x-dtu-chaos", "rate_limit_429") |> Router.call(@opts)
    assert conn.status == 429
    assert {"retry-after", "60"} in conn.resp_headers
  end

  test "chaos outage_503 returns 503" do
    conn = conn(:get, "/repos/o/r/issues") |> put_req_header("x-dtu-chaos", "outage_503") |> Router.call(@opts)
    assert conn.status == 503

  test "chaos unsupported value returns 400 with enum list" do
    conn = conn(:get, "/repos/o/r/issues") |> put_req_header("x-dtu-chaos", "weird") |> Router.call(@opts)
    assert conn.status == 400

defmodule Kiln.Sandboxes.DTU.HealthPoll do
  @moduledoc """
  Polls DTU sidecar `/healthz` every 30s (D-126). After 3 consecutive
  misses: emits `dtu_health_degraded` audit + broadcasts `{:dtu_unhealthy, reason}`
  on `Kiln.PubSub` topic `"dtu_health"` (Phase 7 LiveView consumes).
  """

  use GenServer
  require Logger
  alias Kiln.Audit

  @poll_interval_ms :timer.seconds(30)
  @degrade_threshold 3
  @dtu_url "http://172.28.0.10:80/healthz"

  defstruct misses: 0, last_success_at: nil

  @spec start_link(keyword()) :: GenServer.on_start()
  def start_link(opts), do: GenServer.start_link(__MODULE__, opts, name: __MODULE__)

  @impl true
  def init(_opts) do
    send(self(), :poll)
    {:ok, %__MODULE__{}}
  end

  @impl true
  def handle_info(:poll, state) do
    new_state = do_poll(state)
    Process.send_after(self(), :poll, @poll_interval_ms)
    {:noreply, new_state}
  end

  @impl true
  def handle_info(_, state), do: {:noreply, state}

  defp do_poll(state) do
    case Req.get(@dtu_url, finch: Kiln.Finch, receive_timeout: 5_000) do
      {:ok, %{status: 200}} ->
