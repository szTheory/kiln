## `Kiln.Logger.Metadata.with_metadata/2` (D-45 block API)

```elixir
defmodule Kiln.Logger.Metadata do
  @moduledoc """
  Block-style decorator for scoping Logger metadata to a function body
  (synchronous code paths). Resets metadata on block exit, even if the
  block raises.

      Kiln.Logger.Metadata.with_metadata([run_id: rid, stage_id: sid], fn ->
        Logger.info("stage started")
      end)

  For cross-process metadata propagation, use `Kiln.Telemetry.pack_ctx/0` +
  `Kiln.Telemetry.unpack_ctx/1` (D-45 explicit API).
  """

  @spec with_metadata(keyword(), (() -> any())) :: any()
  def with_metadata(meta, fun) when is_list(meta) and is_function(fun, 0) do
    prior = Logger.metadata()
    try do
      Logger.metadata(meta)
      fun.()
    after
      Logger.reset_metadata(prior)
    end
  end
end
```

## `Kiln.Telemetry.pack_ctx/0` + `unpack_ctx/1` + `async_stream/3` (D-45)

```elixir
defmodule Kiln.Telemetry do
  @moduledoc """
  Cross-process context propagation. Pack current Logger metadata + OTel span ctx
  on the caller side; unpack in the spawned process/worker.
  """

  @metadata_keys [:correlation_id, :causation_id, :actor, :actor_role, :run_id, :stage_id]

  @spec pack_ctx() :: %{String.t() => term()}
  def pack_ctx do
    meta = Logger.metadata()

    @metadata_keys
    |> Enum.into(%{}, fn k -> {Atom.to_string(k), meta[k] || :none} end)
  end

  @spec unpack_ctx(map()) :: :ok
  def unpack_ctx(ctx) when is_map(ctx) do
    meta =
      @metadata_keys
      |> Enum.map(fn k ->
        v = Map.get(ctx, Atom.to_string(k), :none)
        {k, normalize(v)}
      end)

    Logger.metadata(meta)
    :ok
  end

  defp normalize("none"), do: :none
  defp normalize(v), do: v

  @doc """
  Wrapper around `Task.async_stream/3` that pre-packs current Logger metadata
  into each task's closure and unpacks inside the child process. Child log
  lines will carry the parent's metadata without explicit threading at the
  call site.
  """
  @spec async_stream(Enumerable.t(), (term() -> term()), keyword()) :: Enumerable.t()
  def async_stream(enumerable, fun, opts \\ []) do
    ctx = pack_ctx()

    Task.async_stream(
      enumerable,
      fn item ->
        unpack_ctx(ctx)
        fun.(item)
      end,
      opts
    )
  end

  @doc """
  Enqueue-time helper: wraps an Oban job's args with a `kiln_ctx` meta key
  that the ObanHandler will unpack on perform. Usage:

      %{some: "args"}
      |> MyWorker.new(meta: Kiln.Telemetry.pack_meta())
      |> Oban.insert()
  """
  @spec pack_meta() :: %{String.t() => map()}
  def pack_meta do
    %{"kiln_ctx" => pack_ctx()}
  end
end
```

## `Kiln.Telemetry.ObanHandler` — telemetry attachment

```elixir
defmodule Kiln.Telemetry.ObanHandler do
  @moduledoc """
  Telemetry handler for `[:oban, :job, :start]` and `[:oban, :job, :stop]`
  events. On :start, restores Logger.metadata from the job's kiln_ctx meta
  (set by `Kiln.Telemetry.pack_meta/0` at enqueue time).

  This is what makes behavior LOG-02 true: an Oban job's log lines carry
  the enqueueing process's correlation_id.
  """

  @handler_id {__MODULE__, :oban_job_lifecycle}

  @spec attach() :: :ok | {:error, :already_exists}
  def attach do
    :telemetry.attach_many(
      @handler_id,
      [
        [:oban, :job, :start],
        [:oban, :job, :stop]
      ],
      &__MODULE__.handle_event/4,
      nil
    )
  end

  def handle_event([:oban, :job, :start], _measurements, %{job: job}, _config) do
    case job.meta do
      %{"kiln_ctx" => ctx} when is_map(ctx) -> Kiln.Telemetry.unpack_ctx(ctx)
      _ -> :ok
    end
  end

  def handle_event([:oban, :job, :stop], _m, _meta, _c), do: :ok
end
```

### `test/support/logger_capture_helper.ex`:

```elixir
defmodule Kiln.LoggerCaptureHelper do
  @moduledoc """
  Helpers for tests that need to assert structured metadata on log lines.
  Wraps ExUnit.CaptureLog and decodes each line as JSON.
  """

  import ExUnit.CaptureLog

  @spec capture_json((-> any())) :: {any(), [map()]}
  def capture_json(fun) do
    {log, result} =
      with_captured_result(fn ->
        capture_log(fun)
      end)

    lines =
      log
      |> String.split("\n", trim: true)
      |> Enum.flat_map(fn line ->
        case Jason.decode(line) do
          {:ok, map} -> [map]
          _ -> []
        end
      end)

    {result, lines}
  end

  defp with_captured_result(fun) do
    ref = make_ref()
    me = self()

    log =
      capture_log(fn ->
        result = fun.()
        send(me, {ref, result})
      end)

    receive do
      {^ref, result} -> {log, result}
    after
      0 -> {log, nil}
    end
  end
end
```

defmodule Kiln.Telemetry.MetadataThreadingTest do
  @moduledoc """
  D-47 / Phase 1 Success Criterion #3: a contrived multi-process test proving
  metadata threads across Task.async_stream AND Oban job boundaries.

  This test is the load-bearing proof that OBS-01 is mechanically true, not
  a vibes-based claim.
  """
  use Kiln.DataCase
  use Oban.Testing, repo: Kiln.Repo

  import Kiln.LoggerCaptureHelper
  require Logger
  alias Kiln.Telemetry

  defmodule ProbeWorker do
    use Kiln.Oban.BaseWorker

    @impl Oban.Worker
    def perform(%Oban.Job{args: %{"probe" => probe_id}}) do
      require Logger
      Logger.info("probe_oban_worker running probe=#{probe_id}")
      :ok
    end
  end

  test "LOG-01: Task.async_stream child carries parent's correlation_id" do
    parent_cid = Ecto.UUID.generate()

    {_, lines} =
      capture_json(fn ->
        Logger.metadata(correlation_id: parent_cid)
        Logger.info("parent_line")

        [1, 2, 3]
        |> Telemetry.async_stream(fn i ->
          Logger.info("child_line i=#{i}")
          i * 2
        end)
        |> Enum.to_list()
      end)

    child_lines = Enum.filter(lines, &(get_message(&1) =~ ~r/child_line/))
    assert length(child_lines) == 3

    for line <- child_lines do
      assert get_metadata(line, "correlation_id") == parent_cid,
             "child line missing or wrong correlation_id: #{inspect(line)}"
    end
  end

  test "LOG-02: Oban worker inherits enqueueing process's correlation_id" do
    parent_cid = Ecto.UUID.generate()
    probe = "probe_#{System.unique_integer([:positive])}"

    {_, lines} =
      capture_json(fn ->
        Logger.metadata(correlation_id: parent_cid)

        {:ok, _job} =
          %{probe: probe, idempotency_key: "test:#{probe}"}
          |> ProbeWorker.new(meta: Telemetry.pack_meta())
          |> Oban.insert()

        # Oban.Testing runs jobs synchronously in :inline mode; if using :manual,
        # call Oban.drain_queue here.
        Oban.drain_queue(queue: :default)
      end)

    worker_lines =
      Enum.filter(lines, &(get_message(&1) =~ probe))

    assert length(worker_lines) >= 1,
           "no log line matched probe=#{probe}; lines=#{inspect(lines)}"

    for line <- worker_lines do
      assert get_metadata(line, "correlation_id") == parent_cid,
             "Oban worker log line missing or wrong correlation_id: #{inspect(line)}"
    end
  end

  test "D-47 combined: Task.async_stream + Oban job BOTH carry parent correlation_id" do
    parent_cid = Ecto.UUID.generate()
    probe = "combo_#{System.unique_integer([:positive])}"

    {_, lines} =
      capture_json(fn ->
        Logger.metadata(correlation_id: parent_cid)

        # Path 1: Task.async_stream
        task_work =
          [:a, :b]
          |> Telemetry.async_stream(fn letter ->
            Logger.info("async_stream letter=#{letter}")
            letter
          end)
          |> Enum.to_list()

        # Path 2: Oban job
        {:ok, _} =
          %{probe: probe, idempotency_key: "combo:#{probe}"}
          |> ProbeWorker.new(meta: Telemetry.pack_meta())
          |> Oban.insert()

        Oban.drain_queue(queue: :default)

        task_work
      end)

    async_lines = Enum.filter(lines, &(get_message(&1) =~ "async_stream"))
    oban_lines = Enum.filter(lines, &(get_message(&1) =~ probe))

    assert async_lines != [], "async_stream produced no log lines"
    assert oban_lines != [], "Oban worker produced no log lines"

    for line <- async_lines ++ oban_lines do
      assert get_metadata(line, "correlation_id") == parent_cid
    end
  end

  defp get_metadata(line, key), do: line[key] || get_in(line, ["metadata", key])
  defp get_message(line), do: line["message"] || line[:message] || ""
end
