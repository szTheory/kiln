```elixir
defmodule Kiln.Workflows.Graph do
  @spec topological_sort([%{id: String.t(), depends_on: [String.t()]}]) ::
          {:ok, [String.t()]}
          | {:error, :cycle}
          | {:error, {:missing_dep, String.t()}}
  def topological_sort(stages) do
    g = :digraph.new([:acyclic])

    ids = MapSet.new(stages, & &1.id)
    Enum.each(stages, fn s -> :digraph.add_vertex(g, s.id) end)

    result =
      Enum.reduce_while(stages, :ok, fn s, :ok ->
        Enum.reduce_while(s.depends_on, :ok, fn dep, :ok ->
          cond do
            not MapSet.member?(ids, dep) ->
              {:halt, {:error, {:missing_dep, dep}}}
            match?({:error, _}, :digraph.add_edge(g, dep, s.id)) ->
              {:halt, {:error, :cycle}}
            true -> {:cont, :ok}
          end
        end)
        |> case do
          :ok -> {:cont, :ok}
          other -> {:halt, other}
        end
      end)

    try do
      case result do
        :ok ->
          case :digraph_utils.topsort(g) do
            false -> {:error, :cycle}
            sorted when is_list(sorted) -> {:ok, sorted}
          end
        error -> error
      end
    after
      :digraph.delete(g)  # MANDATORY — :digraph is ETS-backed
    end
  end
end
```

```elixir
defmodule Kiln.Workflows.Loader do
  alias Kiln.Workflows.{Compiler, SchemaRegistry}

  @spec load(Path.t()) :: {:ok, Kiln.Workflows.CompiledGraph.t()} | {:error, term()}
  def load(path) do
    with {:ok, raw} <- read_yaml(path),
         {:ok, _} <- validate_schema(raw),
         {:ok, compiled} <- Compiler.compile(raw) do
      {:ok, compiled}
    end
  end

  defp read_yaml(path) do
    case YamlElixir.read_from_file(path) do
      {:ok, map} when is_map(map) -> {:ok, map}
      {:ok, other} -> {:error, {:yaml_parse, {:not_a_map, other}}}
      {:error, err} -> {:error, {:yaml_parse, err}}
    end
  end

  defp validate_schema(raw) do
    {:ok, root} = SchemaRegistry.fetch(:workflow)
    case JSV.validate(raw, root) do
      {:ok, _} -> {:ok, raw}
      {:error, err} -> {:error, {:schema_invalid, JSV.normalize_error(err)}}
    end
  end
end
```

    ```elixir
    defmodule Kiln.Workflows.CompiledGraph do
      @moduledoc """
      In-memory representation of a validated + topologically-sorted workflow.
      Produced by `Kiln.Workflows.Compiler.compile/1`. Downstream consumers:
        * `Kiln.Runs.RunDirector` — checksum assertion on rehydration (D-94)
        * `Kiln.Stages.StageWorker` — per-stage dispatch lookup by id (Plan 08)
      """

      @type stage :: %{
        required(:id) => String.t(),
        required(:kind) => atom(),
        required(:agent_role) => atom(),
        required(:depends_on) => [String.t()],
        required(:timeout_seconds) => pos_integer(),
        required(:retry_policy) => map(),
        required(:sandbox) => atom(),
        optional(:model_preference) => String.t(),
        optional(:on_failure) => :escalate | %{action: :route, to: String.t(), attach: String.t()}
      }

      @type t :: %__MODULE__{
        id: String.t(),
        version: pos_integer(),
        api_version: String.t(),
        metadata: map(),
        caps: map(),
        model_profile: atom(),
        stages: [stage()],          # topologically sorted
        stages_by_id: %{String.t() => stage()},
        entry_node: String.t(),
        checksum: String.t()         # 64-char lowercase hex sha256
      }

      @enforce_keys [:id, :version, :api_version, :stages, :stages_by_id, :entry_node,
                     :caps, :model_profile, :checksum]

      defstruct [
        :id, :version, :api_version, :stages, :stages_by_id, :entry_node,
        :caps, :model_profile, :checksum,
        metadata: %{}
      ]
    end
    ```

    ```elixir
    defmodule Kiln.Workflows.Compiler do
      @moduledoc """
      Transforms a JSV-validated workflow map into a %CompiledGraph{}. Runs 6
      D-62 Elixir-side validators that JSV cannot express (structural graph
      properties + v1 signature invariant).
      """

      alias Kiln.Stages.ContractRegistry
      alias Kiln.Workflows.{CompiledGraph, Graph}

      @type error_reason ::
              :no_entry_node
              | :multiple_entry_nodes
              | :cycle
              | {:missing_dep, String.t()}
              | {:unknown_kind, atom()}
              | {:on_failure_forward_edge, String.t(), String.t()}
              | :signature_must_be_null

      @spec compile(map()) :: {:ok, CompiledGraph.t()} | {:error, {:graph_invalid, error_reason(), map()}}
      def compile(raw) when is_map(raw) do
        with :ok <- validate_signature_null(raw),
             {:ok, stages_raw} <- fetch_stages(raw),
             {:ok, normalized} <- normalize_stages(stages_raw),
             :ok <- validate_single_entry(normalized),
             :ok <- validate_all_kinds_have_contracts(normalized),
             {:ok, sorted_ids} <- Graph.topological_sort(normalized) |> wrap_graph_error(),
             :ok <- validate_on_failure_ancestors(normalized, sorted_ids),
             {:ok, entry_id} <- find_entry(normalized) do
          stages_ordered = Enum.map(sorted_ids, fn id ->
            Enum.find(normalized, &(&1.id == id))
          end)

          stages_by_id = Map.new(normalized, &{&1.id, &1})

          struct = %CompiledGraph{
            id: raw["id"],
            version: raw["version"],
            api_version: raw["apiVersion"],
            metadata: raw["metadata"] || %{},
            caps: raw["spec"]["caps"],
            model_profile: String.to_existing_atom(raw["spec"]["model_profile"]),
            stages: stages_ordered,
            stages_by_id: stages_by_id,
            entry_node: entry_id,
            checksum: nil  # set below
          }

          {:ok, %{struct | checksum: compute_checksum(struct)}}
        end
      end

      # -- D-62 validators --

      defp validate_signature_null(%{"signature" => nil}), do: :ok
      defp validate_signature_null(%{"signature" => _}) do
        {:error, {:graph_invalid, :signature_must_be_null, %{}}}
      end
      defp validate_signature_null(_), do: :ok  # key absent is treated as null-equivalent

      defp fetch_stages(%{"spec" => %{"stages" => stages}}) when is_list(stages), do: {:ok, stages}
      defp fetch_stages(_), do: {:error, {:graph_invalid, :no_entry_node, %{reason: "missing spec.stages"}}}

      defp normalize_stages(stages_raw) do
        normalized =
          Enum.map(stages_raw, fn s ->
            %{
              id: s["id"],
              kind: String.to_existing_atom(s["kind"]),
              agent_role: String.to_existing_atom(s["agent_role"]),
              depends_on: s["depends_on"] || [],
              timeout_seconds: s["timeout_seconds"],
              retry_policy: s["retry_policy"],
              sandbox: String.to_existing_atom(s["sandbox"]),
              model_preference: s["model_preference"],
              on_failure: normalize_on_failure(s["on_failure"])
            }
          end)
        {:ok, normalized}
      end

      defp normalize_on_failure(nil), do: nil
      defp normalize_on_failure("escalate"), do: :escalate
      defp normalize_on_failure(%{"action" => "route", "to" => to, "attach" => attach}) do
        %{action: :route, to: to, attach: attach}
      end

      defp validate_single_entry(stages) do
        entry_count = Enum.count(stages, fn s -> s.depends_on == [] end)
        cond do
          entry_count == 0 -> {:error, {:graph_invalid, :no_entry_node, %{}}}
          entry_count > 1 -> {:error, {:graph_invalid, :multiple_entry_nodes, %{count: entry_count}}}
          true -> :ok
        end
      end

      defp validate_all_kinds_have_contracts(stages) do
        case Enum.find(stages, fn s ->
               case ContractRegistry.fetch(s.kind) do
                 {:ok, _} -> false
                 _ -> true
               end
             end) do
          nil -> :ok
          stage -> {:error, {:graph_invalid, {:unknown_kind, stage.kind}, %{stage_id: stage.id}}}
        end
      end

      defp wrap_graph_error({:ok, _} = ok), do: ok
      defp wrap_graph_error({:error, :cycle}), do: {:error, {:graph_invalid, :cycle, %{}}}
      defp wrap_graph_error({:error, {:missing_dep, dep}}) do
        {:error, {:graph_invalid, {:missing_dep, dep}, %{}}}
      end

      defp validate_on_failure_ancestors(stages, sorted_ids) do
        # Build index: topological position of each stage id
        pos = sorted_ids |> Enum.with_index() |> Map.new()

        offender =
          Enum.find(stages, fn s ->
            case s.on_failure do
              %{to: to_id} -> Map.get(pos, to_id, -1) >= Map.get(pos, s.id, -1)
              _ -> false
            end
          end)

        case offender do
          nil -> :ok
          s -> {:error, {:graph_invalid, :on_failure_forward_edge, %{from: s.id, to: s.on_failure.to}}}
        end
      end

      defp find_entry(stages) do
        case Enum.find(stages, fn s -> s.depends_on == [] end) do
          nil -> {:error, {:graph_invalid, :no_entry_node, %{}}}
          s -> {:ok, s.id}
        end
      end

      defp compute_checksum(%CompiledGraph{} = g) do
        # Canonical term for checksum: {id, version, api_version, model_profile, caps, stages}
        # Use :erlang.term_to_binary with `sort_maps: true` for determinism; then sha256.
        term = {g.id, g.version, g.api_version, g.model_profile, g.caps, normalize_stages_for_hash(g.stages)}
        bin = :erlang.term_to_binary(term, [:deterministic])
        :crypto.hash(:sha256, bin) |> Base.encode16(case: :lower)
      end

      defp normalize_stages_for_hash(stages) do
        Enum.map(stages, fn s ->
          {s.id, s.kind, s.agent_role, s.depends_on, s.timeout_seconds, s.retry_policy,
           s.sandbox, s.model_preference, s.on_failure}
        end)
      end
    end
    ```

    ```elixir
        stages = [
          %{id: "a", depends_on: []},
          %{id: "b", depends_on: ["a"]},
          %{id: "c", depends_on: ["b"]}
        ]
        assert {:ok, ["a", "b", "c"]} = Graph.topological_sort(stages)
      end

      test "diamond a -> {b, c} -> d produces a valid topological order" do
        stages = [
          %{id: "a", depends_on: []},
          %{id: "b", depends_on: ["a"]},
          %{id: "c", depends_on: ["a"]},
          %{id: "d", depends_on: ["b", "c"]}
        ]
        {:ok, sorted} = Graph.topological_sort(stages)
        a_pos = Enum.find_index(sorted, &(&1 == "a"))
        d_pos = Enum.find_index(sorted, &(&1 == "d"))
        assert a_pos < d_pos
        assert Enum.find_index(sorted, &(&1 == "b")) < d_pos
        assert Enum.find_index(sorted, &(&1 == "c")) < d_pos
      end

      test "cycle a -> b -> a returns {:error, :cycle}" do

        ]
    ```

  - Vector: Plan 01 already handled this — schemas built at compile time; malicious schemas fail build.
  - Mitigation: Plan 05 only CONSUMES the pre-built SchemaRegistry; no runtime schema loading.
  - Residual risk: N/A for this plan.

- **T3: on_failure forward-edge bypass via topologically-equal position** (severity: low)
  - Vector: Two stages with no ordering constraint between them could have `on_failure.to` pointing at the "parallel sibling" — the topological position comparison may allow it.
  - Mitigation: `validate_on_failure_ancestors/2` checks `pos[to] >= pos[from]` — strictly less-than is the valid ancestor relation. A parallel sibling would have the same pos index but different id, so `pos[to] >= pos[from]` catches equal-position references (`:ok` only when strictly less). Test case in compiler_test.exs covers this.
  - Residual risk: None for single-entry-node DAGs (what D-62 validator 1 enforces).

- **T4: Checksum collision allows workflow-swap attack** (severity: low)
  - Vector: Attacker modifies `priv/workflows/<id>.yaml` so its compiled-graph checksum equals the one recorded in `runs.workflow_checksum` — `RunDirector` rehydrates silently onto modified graph.
  - Mitigation: SHA-256 + `:erlang.term_to_binary` with `:deterministic` option produces a cryptographically strong digest over every shape-significant field (D-94's integrity mechanism).
