```elixir
defmodule Kiln.Stages.StageWorker do
  @moduledoc """
  Dispatches a single stage of a run. Enqueued with args:
    %{
      "idempotency_key" => "run:<run_id>:stage:<stage_run_id>",
      "run_id" => <run_id>,
      "stage_run_id" => <stage_run_id>,
      "stage_kind" => "planning" | "coding" | "testing" | "verifying" | "merge",
      "stage_input" => %{...}   # must validate against priv/stage_contracts/v1/<kind>.json
    }
  and meta containing kiln_ctx (packed by Kiln.Telemetry.pack_ctx/0).

  Phase 2 stubs the agent dispatch — happy path produces a canned artifact
  and transitions the run to the next state. Phase 3 replaces the stub.
  """

  use Kiln.Oban.BaseWorker, queue: :stages

  alias Kiln.{Artifacts, Runs}
  alias Kiln.Runs.Transitions
  alias Kiln.Stages.ContractRegistry

  require Logger

  @impl Oban.Worker
  def perform(%Oban.Job{args: args, meta: meta}) do
    # Restore correlation_id from Oban meta (P1 pattern).
    _ = Kiln.Telemetry.unpack_ctx(meta["kiln_ctx"] || %{})

    key = args["idempotency_key"]
    run_id = args["run_id"]
    stage_run_id = args["stage_run_id"]
    stage_kind = String.to_existing_atom(args["stage_kind"])
    stage_input = args["stage_input"] || %{}

    with {:ok, root} <- ContractRegistry.fetch(stage_kind),
         :ok <- validate_input(stage_input, root),
         {_status, op} <- fetch_or_record_intent(key, %{
           op_kind: "stage_dispatch",
           intent_payload: args,
           run_id: run_id,
           stage_id: stage_run_id
         }),
         :ok <- guard_not_completed(op),
         {:ok, _artifact} <- stub_dispatch(run_id, stage_run_id, stage_kind),
         {:ok, _run} <- transition_after_stage(run_id, stage_kind) do
      complete_op(op, %{"result" => "stub_ok", "stage_kind" => args["stage_kind"]})
      :ok
    else
      {:error, {:stage_input_rejected, err}} ->
        _ = Kiln.Audit.append(%{
          event_kind: :stage_input_rejected,
          run_id: run_id,
          stage_id: stage_run_id,
          correlation_id: Logger.metadata()[:correlation_id] || Ecto.UUID.generate(),
          payload: %{
            "stage_run_id" => stage_run_id,
            "stage_kind" => args["stage_kind"],
            "errors" => err
          }
        })
        _ = Transitions.transition(run_id, :escalated, %{reason: :invalid_stage_input})
        {:cancel, {:stage_input_rejected, err}}

      {:error, :already_completed} ->
        :ok  # idempotent: already done

      {:error, reason} ->
        Logger.error("StageWorker failure", run_id: run_id, stage_run_id: stage_run_id, reason: inspect(reason))
        {:error, reason}
    end
  end

  defp validate_input(input, root) do
    case JSV.validate(input, root) do
      {:ok, _} -> :ok
      {:error, err} -> {:error, {:stage_input_rejected, JSV.normalize_error(err)}}
    end
  end

  defp guard_not_completed(%{state: :completed}), do: {:error, :already_completed}
  defp guard_not_completed(_op), do: :ok

  # P2 stub: produce a canned artifact for the stage kind and write it.
  defp stub_dispatch(run_id, stage_run_id, stage_kind) do
    body = ["# Stub output for stage_kind=", Atom.to_string(stage_kind), "\n"]
    Artifacts.put(stage_run_id, "#{stage_kind}.md", body,
      run_id: run_id,
      content_type: "text/markdown",
      producer_kind: Atom.to_string(stage_kind))
  end

  # Transition run to the next state per D-87 matrix. Stub mapping:
  # planning → coding; coding → testing; testing → verifying; verifying → merged; merge → merged.
  defp transition_after_stage(run_id, :planning), do: Transitions.transition(run_id, :coding)
  defp transition_after_stage(run_id, :coding), do: Transitions.transition(run_id, :testing)
  defp transition_after_stage(run_id, :testing), do: Transitions.transition(run_id, :verifying)
  defp transition_after_stage(run_id, :verifying), do: Transitions.transition(run_id, :merged)
  defp transition_after_stage(run_id, :merge), do: Transitions.transition(run_id, :merged)
end
```

    ```elixir
          "stage_input" => input

      end
    ```

    ```elixir
    defmodule Kiln.Integration.WorkflowEndToEndTest do
      use Kiln.ObanCase, async: false
      require Logger

      alias Kiln.{Audit, Repo, Workflows}
      alias Kiln.Runs.Run
      alias Kiln.Stages.{StageRun, StageWorker}
      alias Kiln.Factory.Run, as: RunFactory
      alias Kiln.Factory.StageRun, as: StageRunFactory
      alias Kiln.CasTestHelper

      @moduletag :integration

      setup do
        CasTestHelper.setup_tmp_cas(%{})
        cid = Ecto.UUID.generate()
        Logger.metadata(correlation_id: cid)
        on_exit(fn -> Logger.metadata(correlation_id: nil) end)

        unless Process.whereis(Kiln.Policies.StuckDetector) do
          {:ok, _} = Kiln.Policies.StuckDetector.start_link([])
        end

        {:ok, correlation_id: cid}
      end

      test "5 stages of elixir_phoenix_feature.yaml drive a run queued → merged", %{correlation_id: cid} do
        # Load the real workflow
        {:ok, cg} = Workflows.load("priv/workflows/elixir_phoenix_feature.yaml")

        # Insert run in :queued with matching checksum
        run = RunFactory.insert(:run,
          state: :queued,
          workflow_id: cg.id,
          workflow_version: cg.version,
          workflow_checksum: cg.checksum,
          model_profile_snapshot: %{"role" => "planner"},
          caps_snapshot: cg.caps
        )

        # Drive: first transition to :planning (normally the Intents kickoff does this)
        {:ok, _} = Kiln.Runs.Transitions.transition(run.id, :planning)

        # Then for each of 5 stages, dispatch the StageWorker
        for stage <- cg.stages do
          sr = StageRunFactory.insert(:stage_run,
            run_id: run.id,
            workflow_stage_id: stage.id,
            kind: stage.kind,
            agent_role: stage.agent_role,
            state: :pending,
            timeout_seconds: stage.timeout_seconds,
            sandbox: stage.sandbox
          )

          args = %{
            "idempotency_key" => "run:#{run.id}:stage:#{sr.id}",
            "run_id" => run.id,
            "stage_run_id" => sr.id,
            "stage_kind" => Atom.to_string(stage.kind),
            "stage_input" => build_stage_input(run, sr, stage.kind)
          }

          assert :ok = perform_job(StageWorker, args)
        end

        # Final run state — depends on the stub's transition mapping; for the
        # 5-stage workflow plan→code→test→verify→merge, the expected final
        # state is :merged (from verifying's transition + merge's transition).
        final = Repo.get!(Run, run.id)
        assert final.state == :merged

        # Assert stage_runs count
        assert Repo.aggregate(StageRun |> Ecto.Query.where([s], s.run_id == ^run.id), :count) == 5

        # Assert audit events for state transitions (>=5: queued→planning + 4 stage transitions + merged)
        events = Audit.replay(correlation_id: cid)
        transition_events = Enum.filter(events, &(&1.event_kind == :run_state_transitioned))
        assert length(transition_events) >= 5
      end

      defp build_stage_input(run, sr, kind) do
        ref = %{"sha256" => String.duplicate("f", 64), "size_bytes" => 100, "content_type" => "text/markdown"}
        base = %{
          "run_id" => run.id,
          "stage_run_id" => sr.id,
          "attempt" => 1,
          "spec_ref" => ref,
          "budget_remaining" => %{"tokens_usd" => 1.0, "tokens" => 1000, "elapsed_seconds" => 300},
          "model_profile_snapshot" => %{"role" => "planner", "requested_model" => "sonnet-class", "fallback_chain" => []}
        }

        case kind do
          :planning -> Map.merge(base, %{"holdout_excluded" => true, "last_diagnostic_ref" => nil})
          :coding ->   Map.merge(base, %{"holdout_excluded" => true, "plan_ref" => ref})
          :testing ->  Map.merge(base, %{"holdout_excluded" => true, "code_ref" => ref})
          :verifying -> Map.merge(base, %{"test_output_ref" => ref})   # verifying is the holdout-owner; holdout_excluded may be false/omitted
          :merge ->    Map.merge(base, %{"holdout_excluded" => true, "verifier_verdict_ref" => ref})
        end
      end
    ```
