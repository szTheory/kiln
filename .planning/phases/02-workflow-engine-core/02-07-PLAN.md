```elixir
defmodule Kiln.Runs.RunSubtree do
  @moduledoc """
  Per-run supervisor (:one_for_all, :transient). Phase 2 ships with a minimal
  lived child (Task.Supervisor named via Registry) so the ORCH-02 crash-
  isolation integration test can exercise the :one_for_all semantics with a
  real process. Phase 3 swaps the Task.Supervisor for
  Kiln.Agents.SessionSupervisor + Kiln.Sandboxes.Supervisor as per-run children.

  :one_for_all means a coder crash restarts tester and planner alongside —
  consistent with CLAUDE.md "if an agent crashes, the run recovers or escalates."
  :transient means a clean shutdown (normal exit) does not restart.
  """

  use Supervisor

  @spec child_spec(keyword()) :: Supervisor.child_spec()
  def child_spec(opts) do
    run_id = Keyword.fetch!(opts, :run_id)

    %{
      id: {__MODULE__, run_id},
      start: {__MODULE__, :start_link, [opts]},
      restart: :transient,
      shutdown: 5_000,
      type: :supervisor
    }
  end

  @spec start_link(keyword()) :: Supervisor.on_start()
  def start_link(opts) do
    run_id = Keyword.fetch!(opts, :run_id)
    Supervisor.start_link(__MODULE__, opts, name: via(run_id))
  end

  @impl true
  def init(opts) do
    run_id = Keyword.fetch!(opts, :run_id)

    # Phase 2 minimal lived-child: Task.Supervisor named via the run registry.
    # Its only role is to exist (a real pid that can be killed) so
    # test/integration/run_subtree_crash_test.exs can exercise :one_for_all
    # semantics. Phase 3 replaces this child with Kiln.Agents.SessionSupervisor
    # + Kiln.Sandboxes.Supervisor.
    children = [
      {Task.Supervisor, name: {:via, Registry, {Kiln.RunRegistry, {Kiln.Runs.RunSubtree.Tasks, run_id}}}}
    ]

    Supervisor.init(children, strategy: :one_for_all, max_restarts: 3, max_seconds: 5)
  end

  @doc """
  Returns the pid of the lived-child Task.Supervisor for a given run_id, or
  nil if no subtree is alive. Used by ORCH-02 crash-isolation test to find
  a killable process.
  """
  @spec lived_child_pid(Ecto.UUID.t()) :: pid() | nil
  def lived_child_pid(run_id) do
    case Registry.lookup(Kiln.RunRegistry, {Kiln.Runs.RunSubtree.Tasks, run_id}) do
      [{pid, _}] when is_pid(pid) -> pid
      _ -> nil
    end
  end

  defp via(run_id) do
    {:via, Registry, {Kiln.RunRegistry, {__MODULE__, run_id}}}
  end
end
```

```elixir
defmodule Kiln.Runs.RunDirector do
  use GenServer
  require Logger

  alias Kiln.Runs
  alias Kiln.Runs.{RunSupervisor, RunSubtree}

  @periodic_scan_ms 30_000

  @spec start_link(keyword()) :: GenServer.on_start()
  def start_link(opts), do: GenServer.start_link(__MODULE__, opts, name: __MODULE__)

  @impl true
  def init(_opts) do
    # D-92: supervisor boot NEVER blocks on the scan. Defer to self-message.
    send(self(), :boot_scan)
    {:ok, %{monitors: %{}}}  # {pid => {ref, run_id}}
  end

  @impl true
  def handle_info(:boot_scan, state) do
    state = do_scan(state)
    Process.send_after(self(), :periodic_scan, @periodic_scan_ms)
    {:noreply, state}
  end

  def handle_info(:periodic_scan, state) do
    state = do_scan(state)
    Process.send_after(self(), :periodic_scan, @periodic_scan_ms)
    {:noreply, state}
  end

  def handle_info({:DOWN, _ref, :process, pid, reason}, %{monitors: mons} = state) do
    case Map.pop(mons, pid) do
      {nil, _} -> {:noreply, state}
      {{_ref, run_id}, remaining} ->
        Logger.warning("run subtree died; will rehydrate on next scan",
          run_id: run_id, reason: inspect(reason))
        {:noreply, %{state | monitors: remaining}}
    end
  end

  defp do_scan(state) do
    active = Runs.list_active()
    already = MapSet.new(Map.values(state.monitors), fn {_ref, id} -> id end)

    Enum.reduce(active, state, fn run, acc ->
      if MapSet.member?(already, run.id) do
        acc
      else
        case spawn_subtree(run) do
          {:ok, pid} ->
            ref = Process.monitor(pid)
            %{acc | monitors: Map.put(acc.monitors, pid, {ref, run.id})}
          {:error, reason} ->
            Logger.error("failed to spawn run subtree",
              run_id: run.id, reason: inspect(reason))
            acc
        end
      end
    end)
  end

  defp spawn_subtree(run) do
    case assert_workflow_unchanged(run) do
      :ok ->
        DynamicSupervisor.start_child(RunSupervisor, {RunSubtree, run_id: run.id})

      {:error, :workflow_changed} ->
        _ = Kiln.Runs.Transitions.transition(run.id, :escalated, %{reason: :workflow_changed})
        {:error, :workflow_changed}
    end
  end

  defp assert_workflow_unchanged(run) do
    # D-94 — compute the current on-disk workflow checksum and compare.
    path = Path.join(["priv/workflows", "#{run.workflow_id}.yaml"])

    case File.exists?(path) do
      false ->
        Logger.error("workflow file missing for rehydration", run_id: run.id, path: path)
        {:error, :workflow_changed}  # treat missing as changed — force escalation
      true ->
        case Kiln.Workflows.load(path) do
          {:ok, compiled} ->
            if compiled.checksum == run.workflow_checksum, do: :ok, else: {:error, :workflow_changed}
          {:error, _} ->
            {:error, :workflow_changed}
        end
    end
  end
end
```

## Threats
- **T1: RunDirector crash loops exhaust run-subtree budget** (severity: medium)
  - Vector: A bug in `do_scan/1` (e.g., a match failure on an unexpected Run shape) causes RunDirector to crash. Under :one_for_one, it restarts — but the supervisor's default `max_restarts: 3 in 5 seconds` would shut the whole tree down if `do_scan` crashes three times quickly.
  - Mitigation: `do_scan/1` catches per-run failures (`case spawn_subtree(run) do ... {:error, reason} -> Logger.error(...)`) — a single bad run does not crash the director. The overall restart budget is protection against a deeper bug and IS the intended safety valve (crash + diagnostic log > silent bad state).
  - Residual risk: Document in @moduledoc; observability via existing Logger + audit trail.

- **T2: DynamicSupervisor max_children: 10 accidentally blocks run starts** (severity: low)
  - Vector: More than 10 concurrent runs are started; `DynamicSupervisor.start_child/2` returns `{:error, :max_children}`. RunDirector logs error but the run stays in `:queued` forever.
  - Mitigation: D-95 — solo-op ceiling. Periodic scan re-attempts as slots free. Post-P2 Phase 5 (bounded autonomy) may further bound concurrent runs at the Intents layer.
