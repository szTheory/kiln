```elixir
defmodule Kiln.Artifacts.CAS do
  @cas_root Application.compile_env(:kiln, [:artifacts, :cas_root], "priv/artifacts/cas")
  @tmp_root Application.compile_env(:kiln, [:artifacts, :tmp_root], "priv/artifacts/tmp")

  @spec put_stream(Enumerable.t()) :: {:ok, String.t(), non_neg_integer()} | {:error, term()}
  def put_stream(body) do
    File.mkdir_p!(@tmp_root)
    tmp_path = Path.join(@tmp_root, Ecto.UUID.generate())

    File.open!(tmp_path, [:write, :binary, :raw], fn fd ->
      {hash_state, size} =
        Enum.reduce(body, {:crypto.hash_init(:sha256), 0}, fn chunk, {h, sz} ->
          :ok = :file.write(fd, chunk)
          {:crypto.hash_update(h, chunk), sz + IO.iodata_length(chunk)}
        end)

      digest = :crypto.hash_final(hash_state)
      sha_hex = Base.encode16(digest, case: :lower)
      final_path = cas_path(sha_hex)
      File.mkdir_p!(Path.dirname(final_path))

      case File.rename(tmp_path, final_path) do
        :ok ->
          _ = File.chmod(final_path, 0o444)
          {sha_hex, size}
        {:error, reason} ->
          _ = File.rm(tmp_path)
          throw({:rename_failed, reason})
      end
    end)
  catch
    {:rename_failed, reason} -> {:error, {:rename_failed, reason}}
  else
    {sha_hex, size} -> {:ok, sha_hex, size}
  end

  def cas_path(<<aa::binary-size(2), bb::binary-size(2), _::binary>> = sha) do
    Path.join([@cas_root, aa, bb, sha])
  end
end
```

```elixir
def put(stage_run_id, name, body, opts \\ []) do
  content_type = Keyword.fetch!(opts, :content_type)
  run_id = Keyword.fetch!(opts, :run_id)   # MUST come from caller — derived from stage_run in caller

  with {:ok, sha, size} <- CAS.put_stream(body) do
    Repo.transact(fn ->
      cs = Artifact.changeset(%Artifact{}, %{
        stage_run_id: stage_run_id, run_id: run_id,
        name: name, sha256: sha, size_bytes: size,
        content_type: content_type, schema_version: 1, producer_kind: opts[:producer_kind]
      })

      with {:ok, artifact} <- Repo.insert(cs),
           {:ok, _ev} <- Audit.append(%{
             event_kind: :artifact_written,
             run_id: artifact.run_id,
             stage_id: artifact.stage_run_id,
             correlation_id: Logger.metadata()[:correlation_id] || Ecto.UUID.generate(),
             payload: %{"name" => name, "sha256" => sha, "size_bytes" => size, "content_type" => content_type}
           }) do
        {:ok, artifact}
      end
    end)
  end
end
```

    ```elixir
    defmodule Kiln.Artifacts.Artifact do
      use Ecto.Schema
      import Ecto.Changeset

      @type t :: %__MODULE__{}

      @primary_key {:id, :binary_id, autogenerate: false, read_after_writes: true}
      @foreign_key_type :binary_id

      @content_types ~w(text/markdown text/plain application/x-diff application/json text/x-elixir)a

      @derive {Jason.Encoder, only: [:id, :stage_run_id, :run_id, :name, :sha256, :size_bytes, :content_type, :schema_version, :producer_kind, :inserted_at]}

      schema "artifacts" do
        field(:stage_run_id, :binary_id)
        field(:run_id, :binary_id)
        field(:name, :string)
        field(:sha256, :string)
        field(:size_bytes, :integer)
        field(:content_type, Ecto.Enum, values: @content_types)
        field(:schema_version, :integer, default: 1)
        field(:producer_kind, :string)

        # D-81 only inserted_at (NO timestamps() for updated_at)
        timestamps(type: :utc_datetime_usec, updated_at: false)
      end

      @required [:stage_run_id, :run_id, :name, :sha256, :size_bytes, :content_type]
      @optional [:schema_version, :producer_kind]

      def changeset(artifact, attrs) do
        artifact
        |> cast(attrs, @required ++ @optional)
        |> validate_required(@required)
        |> validate_format(:sha256, ~r/^[0-9a-f]{64}$/)
        |> validate_number(:size_bytes, greater_than_or_equal_to: 0, less_than_or_equal_to: 52_428_800)
        |> validate_inclusion(:content_type, @content_types)
        |> foreign_key_constraint(:stage_run_id)
        |> foreign_key_constraint(:run_id)
        |> unique_constraint([:stage_run_id, :name], name: :artifacts_stage_run_name_idx)
        |> check_constraint(:sha256, name: :artifacts_sha256_format)
        |> check_constraint(:content_type, name: :artifacts_content_type_check)
      end

      def content_types, do: @content_types
    end
    ```

    ```elixir
      @moduledoc """
      Raised by Kiln.Artifacts.read!/1 when a CAS blob's re-hash does not match
      the recorded sha256. See D-84.
      """

      defexception [:artifact_id, :expected, :actual, :path, :message]

      @type t :: %__MODULE__{
        artifact_id: Ecto.UUID.t() | nil,
        expected: String.t() | nil,
        actual: String.t() | nil,
        path: String.t() | nil,
        message: String.t() | nil
      }

      @impl true
      def exception(fields) do
        fields = Keyword.new(fields)
        expected = Keyword.get(fields, :expected, "<nil>")
        actual = Keyword.get(fields, :actual, "<nil>")
        artifact_id = Keyword.get(fields, :artifact_id, "<nil>")

        msg = "artifact corruption detected: artifact_id=#{artifact_id} expected=#{expected} actual=#{actual}"

        struct!(__MODULE__, Keyword.put(fields, :message, msg))
      end
    end
    ```

    ```elixir
      use ExUnit.Case, async: false  # put_stream touches fs
      alias Kiln.Artifacts.CAS
      alias Kiln.CasTestHelper

      setup :setup_cas_roots

      test "put_stream returns sha + size; file lands at cas_path" do

        {:ok, sha, size} = CAS.put_stream(body)
        assert String.length(sha) == 64
        assert Regex.match?(~r/^[0-9a-f]{64}$/, sha)
        assert size == 13
        assert File.exists?(CAS.cas_path(sha))
        # Verify the bytes match by reading back + hashing
        bytes = File.read!(CAS.cas_path(sha))
        assert bytes == "hello, world!"

      test "same bytes produce same sha (dedup)" do

        {:ok, sha1, _} = CAS.put_stream(body1)
    ```

    ```elixir
    defmodule Kiln.Artifacts do
      @moduledoc """
      Kiln's 13th bounded context (D-97). Content-addressed artifact storage
      for per-stage outputs (plans, diffs, logs, test outputs).

      Contracts:
      * `put/4` — streams body through SHA-256 + atomic rename into
        `priv/artifacts/cas/<aa>/<bb>/<sha>`, inserts Artifact row + appends
        `:artifact_written` audit event in a single `Repo.transact/2` (D-80).
      * `read!/1` — re-hashes on every open; raises `Kiln.Artifacts.CorruptionError`
        on mismatch, after appending `:integrity_violation` audit event (D-84).
      * `ref_for/1` — returns the exact shape (`sha256`, `size_bytes`,
        `content_type`) the `artifact_ref` sub-schema expects for stage-input
        handoff between stages (D-75).
      """

      import Ecto.Query
      require Logger

      alias Kiln.{Audit, Repo}
      alias Kiln.Artifacts.{Artifact, CAS, CorruptionError}

      @type body :: Enumerable.t()
      @type put_opts :: [
              content_type: atom() | String.t(),
              run_id: Ecto.UUID.t(),
              producer_kind: String.t() | nil
            ]

      @spec put(Ecto.UUID.t(), String.t(), body(), put_opts()) ::
              {:ok, Artifact.t()} | {:error, term()}
      def put(stage_run_id, name, body, opts) do
        content_type = Keyword.fetch!(opts, :content_type)
        run_id = Keyword.fetch!(opts, :run_id)
        producer_kind = Keyword.get(opts, :producer_kind)

        with {:ok, sha, size} <- CAS.put_stream(body) do
          Repo.transact(fn ->
            cs = Artifact.changeset(%Artifact{}, %{
              stage_run_id: stage_run_id,
              run_id: run_id,
              name: name,
              sha256: sha,
              size_bytes: size,
              content_type: normalize_content_type(content_type),
              schema_version: 1,
              producer_kind: producer_kind
            })

            with {:ok, artifact} <- Repo.insert(cs),
                 {:ok, _ev} <-
                   Audit.append(%{
                     event_kind: :artifact_written,
                     run_id: artifact.run_id,
                     stage_id: artifact.stage_run_id,
                     correlation_id: Logger.metadata()[:correlation_id] || Ecto.UUID.generate(),
                     payload: %{
                       "name" => name,
                       "sha256" => sha,
                       "size_bytes" => size,
                       "content_type" => to_string(content_type)
                     }
                   }) do
              {:ok, artifact}
            end
          end)
        end
      end

      @spec get(Ecto.UUID.t(), String.t()) :: {:ok, Artifact.t()} | {:error, :not_found}
      def get(stage_run_id, name) do
        case Repo.one(
               from(a in Artifact,
                 where: a.stage_run_id == ^stage_run_id and a.name == ^name
               )
             ) do
          nil -> {:error, :not_found}
          a -> {:ok, a}
        end
      end

      @spec read!(Artifact.t()) :: binary()
      def read!(%Artifact{sha256: expected} = artifact) do
        path = CAS.cas_path(expected)
        bytes = File.read!(path)
        actual = :crypto.hash(:sha256, bytes) |> Base.encode16(case: :lower)

        if actual != expected do
          # Audit BEFORE raising — put the integrity violation in the ledger.
          _ = Audit.append(%{
            event_kind: :integrity_violation,
            correlation_id: Logger.metadata()[:correlation_id] || Ecto.UUID.generate(),
            payload: %{
              "artifact_id" => artifact.id,
              "expected_sha" => expected,
              "actual_sha" => actual,
              "path" => path
            }
          })

          raise CorruptionError,
            artifact_id: artifact.id,
            expected: expected,
            actual: actual,
            path: path
        end

        bytes
      end

      @spec stream!(Artifact.t()) :: Enumerable.t()
      def stream!(%Artifact{sha256: sha}) do
        # Streaming path skips integrity-on-every-open; consumers use
        # read!/1 when integrity matters. ScrubWorker (Phase 5) covers
        # periodic integrity verification.
        File.stream!(CAS.cas_path(sha), [], 64 * 1024)
      end

      @spec ref_for(Artifact.t()) :: %{sha256: String.t(), size_bytes: non_neg_integer(), content_type: String.t()}
      def ref_for(%Artifact{sha256: sha, size_bytes: size, content_type: ct}) do
        %{sha256: sha, size_bytes: size, content_type: to_string(ct)}
      end

      @spec by_sha(String.t()) :: [Artifact.t()]
      def by_sha(sha) when is_binary(sha) do
        Repo.all(from(a in Artifact, where: a.sha256 == ^sha))
      end

      defp normalize_content_type(ct) when is_atom(ct), do: ct
      defp normalize_content_type(ct) when is_binary(ct), do: String.to_existing_atom(ct)
    end
    ```

<threat_model>
## Threats
- **T1: Path traversal via artifact name** (severity: medium)
  - Vector: An attacker-controlled `name` parameter containing `../` could cause writes outside the CAS hierarchy (e.g., `name: "../../../etc/passwd"`).
  - Mitigation: `name` is only used as a column value in the `artifacts` row — NEVER as a filesystem path component. CAS paths are derived purely from `sha256` hex, which is produced by `:crypto.hash/2` and regex-validated by the Ecto changeset (`^[0-9a-f]{64}$`). Task 1's `cas_path/1` pattern-matches the sha prefix, so non-hex input fails the bit-size guard.
  - Residual risk: Developer writes a future API that uses `name` as a path — caught by review (CAS.cas_path doesn't accept name input).

- **T2: Symlink-based sandbox escape (TOCTOU)** (severity: medium)
  - Vector: A malicious process replaces `priv/artifacts/tmp/<uuid>` with a symlink pointing outside the tree between `File.open!/3` and `File.rename/2`.
  - Mitigation: `File.open!(path, [:write, :binary, :raw], fn fd -> ... end)` opens with write mode, so if the target is a symlink to an unwritable location the open fails. `File.rename/2` on POSIX `rename(2)` operates on paths, not fds — but since the tmp file was created with a UUID name inside a directory only kiln_app has write access to (mode 0700 recommended for priv/artifacts/tmp/), an attacker would need local-user compromise to interpose. In local-dev solo-op this is out of scope; document as a Phase 3+ hardening concern.
  - Residual risk: `priv/artifacts/tmp/` created with default umask (not 0700) — document in Phase 3 concerns.

- **T3: SHA-256 collision used to overwrite blobs** (severity: low)
  - Vector: An attacker crafts an input with SHA-256 collision against a legitimate blob, calls put/4 with malicious content, the atomic rename overwrites the blob in CAS, subsequent `read!/1` returns the malicious bytes.
  - Mitigation: SHA-256 is cryptographically secure (2^128 work). The "attacker has collision" premise means they've broken SHA-256 — outside threat model for v1. Additionally, per POSIX rename(2), overwriting is atomic — no partial state.
  - Residual risk: Theoretical; accept.

- **T4: 50 MB size cap bypass via multiple chunks (size accumulator integer overflow)** (severity: low)
  - Vector: An attacker streams >50 MB in chunks; the size accumulator overflows or the changeset validate_number is checked only after write.
  - Mitigation: Erlang integers are arbitrary-precision (no int overflow). The size-cap is enforced by `Artifact.changeset` via `validate_number/3` with `less_than_or_equal_to: 52_428_800`. A 100 MB write would succeed at the filesystem level but fail the row INSERT, leaving an orphan blob. Phase 5's ScrubWorker will clean those up; P2's risk is disk exhaustion if the attacker can spam large writes.
  - Residual risk: Pre-flight size check at stage-contract boundary (Plan 08's StageWorker) rejects oversized `artifact_ref.size_bytes > 52_428_800` before any blob is written — defense in depth.

- **T5: Audit event schema rejection aborts the entire put/4 transaction, leaving a CAS blob with no row** (severity: low)
  - Vector: If `Audit.append/1` fails JSV validation (e.g., missing required payload field), the `Repo.transact/2` rollback reverses the row INSERT but the blob remains at its CAS path.
  - Mitigation: This is an orphan (blob in CAS without an Artifact row). `Kiln.Artifacts.ScrubWorker` (P5) discovers and removes orphans. Tests in task 2 assert schema pairing; drift surfaces in CI.
  - Residual risk: Orphan storage usage between put failure and next scrub — bounded by 50 MB × write rate; solo-op scale.
</threat_model>
